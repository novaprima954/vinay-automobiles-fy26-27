/**
 * Login user
 */
function login(username, password) {
  try {
    const sheet = getSheet(CONFIG.SHEETS.USERS);
    const data = sheet.getDataRange().getValues();
    
    // Check credentials (skip header row)
    for (let i = 1; i < data.length; i++) {
      const row = data[i];
      // Columns: User ID(0), Username(1), Password(2), Name(3), Role(4), Active(5)
      
      if (row[1] === username && row[2] === password && row[5] === true) {
        // Valid credentials
        const userId = row[0];
        const user = {
          id: userId,
          username: row[1],
          name: row[3],
          role: row[4]
        };
        
        // Create session
        const sessionId = createSession(userId);
        
        // Update last login
        sheet.getRange(i + 1, 8).setValue(new Date());
        
        return {
          success: true,
          user: user,
          sessionId: sessionId
        };
      }
    }
    
    return {
      success: false,
      message: 'Invalid username or password'
    };
    
  } catch (error) {
    return {
      success: false,
      message: 'Login error: ' + error.toString()
    };
  }
}

/**
 * Create session
 */
function createSession(userId) {
  const sessionId = Utilities.getUuid();
  const sheet = getSheet(CONFIG.SHEETS.SESSIONS);
  const now = new Date();
  const expires = new Date(now.getTime() + CONFIG.SESSION_DURATION);
  
  sheet.appendRow([sessionId, userId, now, expires, true]);
  
  return sessionId;
}

/**
 * Validate session
 */
function validateSession(sessionId) {
  try {
    if (!sessionId) {
      return { success: false, message: 'No session ID' };
    }
    
    const sessionSheet = getSheet(CONFIG.SHEETS.SESSIONS);
    const sessionData = sessionSheet.getDataRange().getValues();
    const now = new Date();
    
    // Find session
    for (let i = 1; i < sessionData.length; i++) {
      const row = sessionData[i];
      // Columns: Session ID(0), User ID(1), Created(2), Expires(3), Active(4)
      
      if (row[0] === sessionId && row[4] === true) {
        const expires = new Date(row[3]);
        
        if (expires > now) {
          // Session valid - get user
          const userId = row[1];
          const userSheet = getSheet(CONFIG.SHEETS.USERS);
          const userData = userSheet.getDataRange().getValues();
          
          for (let j = 1; j < userData.length; j++) {
            if (userData[j][0] === userId && userData[j][5] === true) {
              return {
                success: true,
                user: {
                  id: userData[j][0],
                  username: userData[j][1],
                  name: userData[j][3],
                  role: userData[j][4]
                }
              };
            }
          }
        } else {
          return { success: false, message: 'Session expired' };
        }
      }
    }
    
    return { success: false, message: 'Invalid session' };
    
  } catch (error) {
    return {
      success: false,
      message: 'Session validation error: ' + error.toString()
    };
  }
}

/**
 * Logout
 */
function logout(sessionId) {
  try {
    const sheet = getSheet(CONFIG.SHEETS.SESSIONS);
    const data = sheet.getDataRange().getValues();
    
    for (let i = 1; i < data.length; i++) {
      if (data[i][0] === sessionId) {
        sheet.getRange(i + 1, 5).setValue(false);
        return { success: true };
      }
    }
    
    return { success: false, message: 'Session not found' };
    
  } catch (error) {
    return {
      success: false,
      message: 'Logout error: ' + error.toString()
    };
  }
}

/**
 * Check for duplicate receipt number
 */
function checkDuplicateReceipt(sessionId, receiptNo) {
  try {
    const validation = validateSession(sessionId);
    if (!validation.success) {
      return validation;
    }
    
    const sheet = getSheet(CONFIG.SHEETS.DATA);
    const data = sheet.getDataRange().getValues();
    
    // Check if receipt number exists (Column A)
    for (let i = 1; i < data.length; i++) {
      if (data[i][0] === receiptNo) {
        return {
          success: true,
          isDuplicate: true,
          message: 'Receipt number already exists'
        };
      }
    }
    
    return {
      success: true,
      isDuplicate: false
    };
    
  } catch (error) {
    return {
      success: false,
      message: 'Error checking receipt: ' + error.toString()
    };
  }
}

/**
 * Get all users (admin only)
 */
function getUsers(sessionId) {
  try {
    const validation = validateSession(sessionId);
    if (!validation.success) {
      return validation;
    }
    
    if (validation.user.role !== 'admin') {
      return { success: false, message: 'Access denied' };
    }
    
    const sheet = getSheet(CONFIG.SHEETS.USERS);
    const data = sheet.getDataRange().getValues();
    const users = [];
    
    for (let i = 1; i < data.length; i++) {
      users.push({
        id: data[i][0],
        username: data[i][1],
        name: data[i][3],
        role: data[i][4],
        active: data[i][5],
        created: data[i][6]
      });
    }
    
    return { success: true, users: users };
    
  } catch (error) {
    return {
      success: false,
      message: 'Error fetching users: ' + error.toString()
    };
  }
}

/**
 * Save sales entry with all 42 columns
 */
function saveSales(sessionId, salesData) {
  try {
    const validation = validateSession(sessionId);
    if (!validation.success) {
      return validation;
    }
    
    const user = validation.user;
    if (user.role !== 'admin' && user.role !== 'sales') {
      return { success: false, message: 'Access denied' };
    }
    
    const sheet = getSheet(CONFIG.SHEETS.DATA);

    // Block duplicate receipt numbers
    const existingData = sheet.getDataRange().getValues();
    for (let i = 1; i < existingData.length; i++) {
      if (String(existingData[i][0]).trim() === String(salesData.receiptNo).trim()) {
        return { success: false, message: 'Receipt No ' + salesData.receiptNo + ' already exists. Duplicate entry not allowed.' };
      }
    }

    // Handle "Other" financier name
    let financierName = salesData.financierName || '';
    
    Logger.log('Financier Name received: ' + financierName);
    
    // Calculate cash total and grand total
    const r1 = parseFloat(salesData.receipt1Amount) || 0;
    const r2 = parseFloat(salesData.receipt2Amount) || 0;
    const r3 = parseFloat(salesData.receipt3Amount) || 0;
    const r4 = parseFloat(salesData.receipt4Amount) || 0;
    const disbursed = parseFloat(salesData.disbursedAmount) || 0;
    
    const cashTotal = r1 + r2 + r3 + r4;
    const grandTotal = cashTotal + disbursed;
    
    Logger.log('=== PAYMENT AMOUNTS ===');
    Logger.log('R1: ' + r1 + ', R2: ' + r2 + ', R3: ' + r3 + ', R4: ' + r4);
    Logger.log('Disbursed: ' + disbursed);
    Logger.log('Cash Total: ' + cashTotal);
    Logger.log('Grand Total: ' + grandTotal);
    
    // Build row data - EXACTLY 42 columns (A through AP)
    const rowData = [
      salesData.receiptNo || '',                    // 0:  A - Receipt No
      salesData.executiveName || user.name || user.username || '', // 1: B - Executive Name
      salesData.bookingDate || new Date(),          // 2:  C - Date

      salesData.deliveryDate || '',                 // 11: L - Delivery Date
      salesData.guard || '',                        // 12: M - Guard
      salesData.gripcover || salesData.gripCover || '', // 13: N - Grip Cover
      salesData.seatcover || salesData.seatCover || '', // 14: O - Seat Cover
      salesData.matin || '',                        // 15: P - Matin
      salesData.tankcover || salesData.tankCover || '', // 16: Q - Tank Cover
      salesData.handlehook || salesData.handleHook || '', // 17: R - Handle Hook
      salesData.helmet || '',                       // 18: S - Helmet
      salesData.salesRemark || '',                  // 19: T - Sales Remark
      '',                                           // 20: U - Accountant Name
      false,                                        // 21: V - Account Check
      '',                                           // 22: W - Account Remark
      salesData.receiptNo1 || '',                   // 23: X - Receipt No 1
      r1,                                           // 24: Y - Receipt 1 Amount
      salesData.receiptNo2 || '',                   // 25: Z - Receipt No 2
      r2,                                           // 26: AA - Receipt 2 Amount
      salesData.receiptNo3 || '',                   // 27: AB - Receipt No 3
      r3,                                           // 28: AC - Receipt 3 Amount
      salesData.receiptNo4 || '',                   // 29: AD - Receipt No 4
      r4,                                           // 30: AE - Receipt 4 Amount
      salesData.doNumber || '',                     // 31: AF - DO Number
      disbursed,                                    // 32: AG - Disbursed Amount
      cashTotal,                                    // 33: AH - Cash Total
      grandTotal,                                   // 34: AI - Grand Total
      '',                                           // 35: AJ - Accessory Checker Name
      false,                                        // 36: AK - Accessory Fitted
      '',                                           // 37: AL - Remark

/**
 * Helper function to format dates
 */
function formatDate(date) {
  if (!date) return '';
  try {
    if (typeof date === 'string') return date;
    const d = new Date(date);
    const year = d.getFullYear();
    const month = String(d.getMonth() + 1).padStart(2, '0');
    const day = String(d.getDate()).padStart(2, '0');
    return year + '-' + month + '-' + day;
  } catch (e) {
    return date.toString();
  }
}

/**
 * Get accounts dashboard data
 */
function getAccountsDashboard(sessionId, month) {
  try {
    const validation = validateSession(sessionId);
    if (!validation.success) {
      return validation;
    }

    const user = validation.user;
    if (user.role !== 'admin' && user.role !== 'accounts') {
      return { success: false, message: 'Access denied' };
    }

    const sheet = getSheet(CONFIG.SHEETS.DATA);
    const data = sheet.getDataRange().getValues();

    // Parse month (format: YYYY-MM), empty = All Months
    const parts = month ? month.split('-') : [];
    const filterYear  = parts[0] ? parseInt(parts[0]) : null;
    const filterMonth = parts[1] ? parseInt(parts[1]) : null;

    let yesCount = 0;
    let noCount = 0;
    let blankCount = 0;

    // Count records for the selected month (skip header)
    for (let i = 1; i < data.length; i++) {
      const row = data[i];
      if (!row[0]) continue;
      const bookingDate = row[2]; // Column C: Date
      const accountCheck = row[21]; // Column V: Account Check

      // Month filter (skip if month specified and doesn't match)
      if (filterYear && filterMonth && bookingDate) {
        const recordDate = new Date(bookingDate);
        if (recordDate.getFullYear() !== filterYear || (recordDate.getMonth() + 1) !== filterMonth) {
          continue;
        }
      }

      if (accountCheck === 'Yes') {
        yesCount++;
      } else if (accountCheck === 'No') {
        noCount++;
      } else {
        blankCount++;
      }
    }
    
    return {
      success: true,
      dashboard: {
        accountCheckYes: yesCount,
        accountCheckNo: noCount,
        accountCheckBlank: blankCount
      }
    };
    
  } catch (error) {
    Logger.log('Error in getAccountsDashboard: ' + error.toString());
    return {
      success: false,
      message: 'Error loading dashboard: ' + error.toString()
    };
  }
}

/**
 * Search accounts records
 */
function searchAccountsRecords(sessionId, searchBy, searchValue, dateFilter, singleDate, fromDate, toDate) {
  try {
    const validation = validateSession(sessionId);
    if (!validation.success) {
      return validation;
    }
    
    const user = validation.user;
    if (user.role !== 'admin' && user.role !== 'accounts') {
      return { success: false, message: 'Access denied' };
    }
    
    const sheet = getSheet(CONFIG.SHEETS.DATA);
    const data = sheet.getDataRange().getValues();
    
    const results = [];
    
    // Search through data (skip header)
    for (let i = 1; i < data.length; i++) {
      const row = data[i];
      let matches = false;
      
      if (searchBy === 'Receipt No') {
        matches = row[0] && row[0].toString().includes(searchValue);
      } else if (searchBy === 'Customer Name') {
        matches = row[3] && row[3].toString().toLowerCase().includes(searchValue.toLowerCase());
      } else if (searchBy === 'Mobile No') {
        matches = row[4] && row[4].toString().includes(searchValue);
      } else if (searchBy === 'Executive Name') {
        matches = row[1] && row[1].toString().toLowerCase().includes(searchValue.toLowerCase());
      } else if (searchBy === 'Variant') {
        matches = row[6] && row[6].toString().toLowerCase().includes(searchValue.toLowerCase());
      } else if (searchBy === 'Booking Date') {
        const recordDate = new Date(row[2]);
        const today = new Date();
        
        if (dateFilter === 'today') {
          matches = recordDate.toDateString() === today.toDateString();
        } else if (dateFilter === 'last7') {
          const sevenDaysAgo = new Date(today.getTime() - 7 * 24 * 60 * 60 * 1000);
          matches = recordDate >= sevenDaysAgo && recordDate <= today;
        } else if (dateFilter === 'last30') {
          const thirtyDaysAgo = new Date(today.getTime() - 30 * 24 * 60 * 60 * 1000);
          matches = recordDate >= thirtyDaysAgo && recordDate <= today;
        } else if (dateFilter === 'single' && singleDate) {
          const targetDate = new Date(singleDate);
          matches = recordDate.toDateString() === targetDate.toDateString();
        } else if (dateFilter === 'range' && fromDate && toDate) {
          const from = new Date(fromDate);
          const to = new Date(toDate);
          matches = recordDate >= from && recordDate <= to;
        }
      }
      
      if (matches) {
        results.push({
          receiptNo: row[0],
          executiveName: row[1],
          bookingDate: formatDate(row[2]),
          customerName: row[3],
          mobileNo: row[4],
          model: row[5],
          variant: row[6],
          accountCheck: row[21] || 'Blank'
        });
      }
    }
    
    return {
      success: true,
      results: results,
      message: 'Found ' + results.length + ' record(s)'
    };
    
  } catch (error) {
    Logger.log('Error in searchAccountsRecords: ' + error.toString());
    return {
      success: false,
      message: 'Error searching records: ' + error.toString()
    };
  }
}

/**
 * Export accounts to CSV
 */
function exportAccountsToCSV(sessionId, month, status) {
  try {
    const validation = validateSession(sessionId);
    if (!validation.success) {
      return validation;
    }
    
    const user = validation.user;
    if (user.role !== 'admin' && user.role !== 'accounts') {
      return { success: false, message: 'Access denied' };
    }
    
    const sheet = getSheet(CONFIG.SHEETS.DATA);
    const data = sheet.getDataRange().getValues();
    
    // Parse month
    const [year, monthNum] = month.split('-');
    
    let csv = data[0].join(',') + '\n'; // Header row
    
    // Add matching data rows
    for (let i = 1; i < data.length; i++) {
      const row = data[i];
      const bookingDate = row[2];
      const accountCheck = row[21];
      
      if (bookingDate) {
        const recordDate = new Date(bookingDate);
        if (recordDate.getFullYear() == year && (recordDate.getMonth() + 1) == monthNum) {
          let matches = false;
          
          if (status === 'all') {
            matches = true;
          } else if (status === 'yes' && accountCheck === 'Yes') {
            matches = true;
          } else if (status === 'no' && accountCheck === 'No') {
            matches = true;
          } else if (status === 'blank' && (!accountCheck || accountCheck === '')) {
            matches = true;
          }
          
          if (matches) {
            csv += row.map(function(cell) {
              return '"' + (cell || '').toString().replace(/"/g, '""') + '"';
            }).join(',') + '\n';
          }
        }
      }
    }
    
    return {
      success: true,
      csv: csv
    };
    
  } catch (error) {
    Logger.log('Error in exportAccountsToCSV: ' + error.toString());
    return {
      success: false,
      message: 'Error exporting: ' + error.toString()
    };
  }
}

/**
 * Get CRM Dashboard data
 */
function getCRMDashboard(sessionId) {
  try {
    const validation = validateSession(sessionId);
    if (!validation.success) {
      return validation;
    }
    
    const user = validation.user;
    
    // Only admin and sales can access CRM
    if (user.role !== 'admin' && user.role !== 'sales') {
      return {
        success: false,
        message: 'Access denied. CRM is only available to admin and sales users.'
      };
    }
    
    const sheet = getSheet('CRM_Leads');
    const data = sheet.getDataRange().getValues();
    
    let myNew = 0;
    let myHot = 0;
    let myFollowUp = 0;
    let available = 0;
    let converted = 0;
    let totalLeads = 0;
    let urgentFollowUps = [];
    
    const today = new Date();
    today.setHours(0, 0, 0, 0);
    
    // Count leads (skip header)
    for (let i = 1; i < data.length; i++) {
      const row = data[i];
      const status = row[7]; // Column H: Status
      const assignedTo = row[8]; // Column I: Assigned To
      const followUpDate = row[12]; // Column M: Next Follow-up Date
      
      // Available leads (no status, not assigned)
      if (!status || status === '') {
        available++;
        continue;
      }
      
      // Admin sees all, sales sees only their own
      if (user.role === 'admin' || assignedTo === user.name) {
        totalLeads++;

        if (status === 'New') myNew++;
        if (status === 'Hot Lead') myHot++;
        if (status === 'Converted') converted++;
        
        // Check for urgent follow-ups (today)
        if (followUpDate) {
          const followUp = new Date(followUpDate);
          followUp.setHours(0, 0, 0, 0);
          
          if (followUp.getTime() === today.getTime()) {
            myFollowUp++;
            urgentFollowUps.push({
              name: row[1],
              mobile: row[2],
              model: row[5],
              leadId: row[0]
            });
          }
        }
      }
    }
    
    return {
      success: true,
      dashboard: {
        myNew: myNew,
        myHot: myHot,
        myFollowUp: myFollowUp,
        available: available,
        converted: converted,
        totalLeads: totalLeads,
        urgentFollowUps: urgentFollowUps
      }
    };
    
  } catch (error) {
    Logger.log('Error in getCRMDashboard: ' + error.toString());
    return {
      success: false,
      message: 'Error loading dashboard: ' + error.toString()
    };
  }
}

/**
 * Calculate monthly trend for last 6 months (completed sales only)
 */
function calculateMonthlyTrend(data, executiveName) {
  const now = new Date();
  const months = [];
  
  // Generate last 6 months
  for (let i = 5; i >= 0; i--) {
    const d = new Date(now.getFullYear(), now.getMonth() - i, 1);
    months.push({
      year: d.getFullYear(),
      month: d.getMonth(),
      label: d.toLocaleDateString('en-US', { month: 'short', year: 'numeric' }),
      count: 0
    });
  }
  
  // Count completed sales per month
  for (let i = 1; i < data.length; i++) {
    const row = data[i];
    const executive = row[1];
    const bookingDate = row[2];
    const accountCheck = row[21];
    
    if (executive === executiveName && accountCheck === 'Yes' && bookingDate) {
      const date = new Date(bookingDate);
      
      for (let j = 0; j < months.length; j++) {
        if (date.getFullYear() === months[j].year && date.getMonth() === months[j].month) {
          months[j].count++;
          break;
        }
      }
    }
  }
  
  // Calculate percentages (relative to max)
  const maxCount = Math.max.apply(null, months.map(function(m) { return m.count; }).concat([1]));
  
  return months.map(function(m) {
    return {
      month: m.label,
      count: m.count,
      percentage: Math.round((m.count / maxCount) * 100)
    };
  });
}

    }

      const targetValue = targetsData[i][2]; // Column C: Target

      handleHook: 0,
      rainCover: 0,

      const accessoryFitted = row[36]; // Column AK
      const dmsStatus = row[42]; // Column AQ
      const insuranceStatus = row[45]; // Column AT
      const rtoStatus = row[48]; // Column AW
      
      // Only count user's own sales
      if (executive === user.name) {
        mySales++;
        
        // Count completed sales (Account Check = Yes)
        if (accountCheck === 'Yes') {
          myCompletedSales++;
          
          // Model breakdown (only completed sales)
          if (model) {
            if (!modelBreakdownMap[model]) {
              modelBreakdownMap[model] = 0;
            }
            modelBreakdownMap[model]++;
          }
          
          // Count accessories (only when Account Check = Yes AND accessory = Yes)
          if (row[12] === 'Yes') myAccessories.guard++;
          if (row[13] === 'Yes') myAccessories.grip++;
          if (row[14] === 'Yes') myAccessories.seatCover++;
          if (row[15] === 'Yes') myAccessories.matin++;

          if (row[61] === 'Yes') myAccessories.rainCover++;
          if (row[62] === 'Yes') myAccessories.buzzer++;
          if (row[63] === 'Yes') myAccessories.backRest++;

        }
        
        // Count pending tasks (for all sales, not just completed)
        if (dmsStatus !== 'Done') dmsPending++;
        if (insuranceStatus !== 'Done') insurancePending++;
        if (rtoStatus !== 'Done') rtoPending++;
        if (accessoryFitted !== 'Yes') accessoriesPending++;
      }
    }
    
    // Convert model breakdown to array
    const modelBreakdown = Object.keys(modelBreakdownMap).map(function(model) {
      return {
        model: model,
        count: modelBreakdownMap[model]
      };
    }).sort(function(a, b) {
      return b.count - a.count;
    });
    
    // Calculate monthly trend (last 6 months, completed sales only)
    const monthlyTrend = calculateMonthlyTrend(data, user.name);
    
    // Get team comparison
    const teamStats = {};
    for (let i = 1; i < filteredData.length; i++) {
      const row = filteredData[i];
      const executive = row[1];
      const accountCheck = row[21];
      
      if (!teamStats[executive]) {

        };
      }
      
      teamStats[executive].totalSales++;
      
      if (accountCheck === 'Yes') {
        teamStats[executive].completedSales++;
      }
    }
    
    // Convert to sorted array
    const teamComparison = Object.keys(teamStats).map(function(exec) {
      return {
        executive: exec,
        totalSales: teamStats[exec].totalSales,
        completedSales: teamStats[exec].completedSales
      };
    }).sort(function(a, b) {
      return b.completedSales - a.completedSales;
    });
    
    // Calculate ranking
    let myRank = 0;
    for (let i = 0; i < teamComparison.length; i++) {
      if (teamComparison[i].executive === user.name) {
        myRank = i + 1;
        break;
      }
    }
    
    // Calculate target progress
    const targetProgress = target > 0 ? Math.round((myCompletedSales / target) * 100) : 0;
    
    return {
      success: true,
      dashboard: {
        mySales: mySales,
        myCompletedSales: myCompletedSales,
        target: target,
        targetProgress: targetProgress,
        modelBreakdown: modelBreakdown,
        monthlyTrend: monthlyTrend,
        dmsPending: dmsPending,
        insurancePending: insurancePending,
        rtoPending: rtoPending,
        accessoriesPending: accessoriesPending,
        myAccessories: myAccessories,
        myRank: myRank,
        totalExecutives: teamComparison.length,
        teamComparison: teamComparison
      }
    };
    
  } catch (error) {
    Logger.log('Error in getSalesDashboard: ' + error.toString());
    return {
      success: false,
      message: 'Error loading dashboard: ' + error.toString()
    };
  }
}

          receiptNo: receiptNo,
          customerName: customerName,
          model: model,
          accessories: accessoriesText,
          daysAgo: daysAgo
        });
        
        // Overdue if >7 days
        if (daysAgo > 7) {
          issues++;
          overdueList.push({

            daysAgo: daysAgo
          });
        }

        fittedCount++;

 * FIXED: Column indices and checks for Account Check = Yes AND accessory = Yes
 */
function getAccessoryBreakdown(sessionId, accessoryType, dateFilter) {
  try {
    const validation = validateSession(sessionId);
    if (!validation.success) {
      return validation;
    }
    
    const user = validation.user;
    if (user.role !== 'admin') {
      return {
        success: false,
        message: 'Access denied'
      };
    }
    
    const sheet = getSheet(CONFIG.SHEETS.DATA);
    const data = sheet.getDataRange().getValues();
    const filteredData = filterByDateRange(data, dateFilter, 2);
    
    // FIXED: Map accessory type to correct column index (M-S = 12-18)
    const accessoryColumns = {
      'guard': 12,        // Column M
      'grip': 13,         // Column N
      'seatCover': 14,    // Column O
      'matin': 15,        // Column P
      'tankCover': 16,    // Column Q
      'handleHook': 17,   // Column R
      'helmet': 18,        // Column S
      'raincover': 61, 
      'buzzer': 62, 
      'backrest': 63,

    };
    
    const colIndex = accessoryColumns[accessoryType];
    if (!colIndex) {
      return {
        success: false,
        message: 'Invalid accessory type'
      };
    }
    
    const modelBreakdown = {};
    
    // Count by model (skip header)
    // FIXED: ONLY count when Account Check = Yes AND accessory = Yes
    for (let i = 1; i < filteredData.length; i++) {
      const row = filteredData[i];
      const model = row[5]; // Column F
      const accountCheck = row[21]; // Column V
      const hasAccessory = row[colIndex];
      
      // CRITICAL: Only count if Account Check = Yes AND accessory = Yes
      if (accountCheck === 'Yes' && hasAccessory === 'Yes') {
        if (!modelBreakdown[model]) {
          modelBreakdown[model] = 0;
        }
        modelBreakdown[model]++;
      }
    }
    
    // Convert to sorted array
    const breakdown = Object.keys(modelBreakdown).map(model => ({
      model: model,
      count: modelBreakdown[model]
    })).sort((a, b) => b.count - a.count);
    
    return {
      success: true,
      breakdown: breakdown
    };
    
  } catch (error) {
    Logger.log('Error in getAccessoryBreakdown: ' + error.toString());
    return {
      success: false,
      message: 'Error: ' + error.toString()
    };
  }
}

/**
 * Filter data by date range
 */
function filterByDateRange(data, dateFilter, dateColumnIndex) {
  if (!dateFilter || dateFilter === 'all') {
    return data;
  }
  
  const today = new Date();
  today.setHours(0, 0, 0, 0);
  
  const filtered = [data[0]]; // Keep header
  
  for (let i = 1; i < data.length; i++) {
    const row = data[i];
    
    // Skip empty rows
    if (!row || !row[dateColumnIndex]) {
      continue;
    }
    
    const rowDate = new Date(row[dateColumnIndex]);
    
    // Skip invalid dates
    if (isNaN(rowDate.getTime())) {
      continue;
    }
    
    rowDate.setHours(0, 0, 0, 0);
    
    let include = false;
    
    if (dateFilter === 'today') {
      include = rowDate.getTime() === today.getTime();
    } else if (dateFilter === 'week') {
      const weekAgo = new Date(today);
      weekAgo.setDate(weekAgo.getDate() - 7);
      include = rowDate >= weekAgo && rowDate <= today;
    } else if (dateFilter === 'month') {
      const monthAgo = new Date(today);
      monthAgo.setMonth(monthAgo.getMonth() - 1);
      include = rowDate >= monthAgo && rowDate <= today;
    } else {
      include = true;
    }
    
    if (include) {
      filtered.push(row);
    }
  }
  
  return filtered;
}

/**
 * Update operator status
 * FIXED: Remove hierarchy validation (allow updating all at once)
 */
function updateOperatorStatus(sessionId, receiptNo, data) {
  try {
    const validation = validateSession(sessionId);
    if (!validation.success) {
      return validation;
    }
    
    const user = validation.user;
    
    // Only operator and admin can access
    if (user.role !== 'operator' && user.role !== 'admin') {
      return {
        success: false,
        message: 'Access denied'
      };
    }
    
    const sheet = getSheet(CONFIG.SHEETS.DATA);
    const sheetData = sheet.getDataRange().getValues();
    
    Logger.log('=== updateOperatorStatus START ===');
    Logger.log('Receipt: ' + receiptNo);
    Logger.log('User: ' + user.name);
    Logger.log('Data: ' + JSON.stringify(data));
    
    // Find record
    for (let i = 1; i < sheetData.length; i++) {
      const row = sheetData[i];
      
      if (row[0] && row[0].toString() === receiptNo.toString()) {
        const targetRow = i + 1;
        const now = new Date();
        
        // Get current values
        const currentDms = row[42] || '';
        const currentInsurance = row[45] || '';
        const currentVahan = row[48] || '';
        const currentNumberPlateFitted = row[52] || '';
        
        Logger.log('Current Status: DMS=' + currentDms + ', Insurance=' + currentInsurance + ', Vahan=' + currentVahan);
        
        // CRITICAL: Check if ALL are complete (locked state)
        const allComplete = currentDms === 'Yes' && currentInsurance === 'Yes' && currentVahan === 'Yes';
        
        Logger.log('All Complete: ' + allComplete);
        
        if (allComplete) {
          // If all complete, ONLY allow number plate updates
          // Block any attempts to change DMS/Insurance/Vahan/Engine/Chassis
          if (data.dmsStatus || data.insuranceStatus || data.vahanStatus || 
              data.engineNumber || data.frameNumber) {
            Logger.log('❌ BLOCKED: All statuses complete - can only update number plate');
            return {
              success: false,
              message: 'All statuses are complete and locked. Only Number Plate can be updated.'
            };
          }
        }
        
        // Check individual locks
        if (currentDms === 'Yes' && data.dmsStatus && data.dmsStatus !== currentDms) {
          return { success: false, message: 'DMS is locked (already marked Yes)' };
        }
        if (currentInsurance === 'Yes' && data.insuranceStatus && data.insuranceStatus !== currentInsurance) {
          return { success: false, message: 'Insurance is locked (already marked Yes)' };
        }
        if (currentVahan === 'Yes' && data.vahanStatus && data.vahanStatus !== currentVahan) {
          return { success: false, message: 'Vahan is locked (already marked Yes)' };
        }
        if (currentNumberPlateFitted === 'Yes' && 
            (data.numberPlateFitted !== currentNumberPlateFitted || data.numberPlateDetails !== row[51])) {
          return { success: false, message: 'Number Plate is locked (already fitted)' };
        }
        
        // At least one field must be filled
        if (!data.dmsStatus && !data.insuranceStatus && !data.vahanStatus && 
            !data.numberPlateDetails && !data.numberPlateFitted && 
            !data.engineNumber && !data.frameNumber) {
          return { success: false, message: 'At least one field must be filled' };
        }
        
        // WORKFLOW VALIDATION (backend enforcement)
        const newDms = data.dmsStatus || currentDms;
        const newInsurance = data.insuranceStatus || currentInsurance;
        const newVahan = data.vahanStatus || currentVahan;
        
        Logger.log('New Status: DMS=' + newDms + ', Insurance=' + newInsurance + ', Vahan=' + newVahan);
        
        // Rule 1: Insurance=Yes requires DMS=Yes
        if (newInsurance === 'Yes' && newDms !== 'Yes') {
          Logger.log('❌ Workflow violation: Insurance=Yes but DMS≠Yes');
          return {
            success: false,
            message: 'Workflow Error: Insurance can only be "Yes" if DMS is "Yes"'
          };
        }
        
        // Rule 2: Vahan=Yes requires Insurance=Yes
        if (newVahan === 'Yes' && newInsurance !== 'Yes') {
          Logger.log('❌ Workflow violation: Vahan=Yes but Insurance≠Yes');
          return {
            success: false,
            message: 'Workflow Error: Vahan can only be "Yes" if Insurance is "Yes"'
          };
        }
        
        Logger.log('✅ Workflow validation passed');
        
        // Update Engine & Chassis (only if not locked)
        if (!allComplete) {
          if (data.engineNumber !== undefined) {
            sheet.getRange(targetRow, 60).setValue(data.engineNumber); // BH
            Logger.log('Updated Engine: ' + data.engineNumber);
          }
          if (data.frameNumber !== undefined) {
            sheet.getRange(targetRow, 61).setValue(data.frameNumber); // BI
            Logger.log('Updated Chassis: ' + data.frameNumber);
          }
        }
        
        // Update DMS (only if not locked)
        if (data.dmsStatus && currentDms !== 'Yes') {
          sheet.getRange(targetRow, 43).setValue(data.dmsStatus); // AQ
          if (data.dmsStatus === 'Yes') {
            sheet.getRange(targetRow, 44).setValue(now); // AR: Date
            sheet.getRange(targetRow, 45).setValue(user.name); // AS: Operator
            Logger.log('✅ DMS marked Yes by ' + user.name);
          }
        }
        
        // Update Insurance (only if not locked)
        if (data.insuranceStatus && currentInsurance !== 'Yes') {
          sheet.getRange(targetRow, 46).setValue(data.insuranceStatus); // AT
          if (data.insuranceStatus === 'Yes') {
            sheet.getRange(targetRow, 47).setValue(now); // AU: Date
            sheet.getRange(targetRow, 48).setValue(user.name); // AV: Operator
            Logger.log('✅ Insurance marked Yes by ' + user.name);
          }
        }
        
        // Update Vahan (only if not locked)
        if (data.vahanStatus && currentVahan !== 'Yes') {
          sheet.getRange(targetRow, 49).setValue(data.vahanStatus); // AW
          if (data.vahanStatus === 'Yes') {
            sheet.getRange(targetRow, 50).setValue(now); // AX: Date
            sheet.getRange(targetRow, 51).setValue(user.name); // AY: Operator
            Logger.log('✅ Vahan marked Yes by ' + user.name);
          }
        }
        
        // Update Number Plate (always allowed)
        if (data.numberPlateDetails !== undefined) {
          sheet.getRange(targetRow, 52).setValue(data.numberPlateDetails); // AZ
          Logger.log('Updated Number Plate: ' + data.numberPlateDetails);
        }
        if (data.numberPlateFitted && currentNumberPlateFitted !== 'Yes') {
          sheet.getRange(targetRow, 53).setValue(data.numberPlateFitted); // BA
          if (data.numberPlateFitted === 'Yes') {
            sheet.getRange(targetRow, 54).setValue(now); // BB: Date
            sheet.getRange(targetRow, 55).setValue(user.name); // BC: Operator
            Logger.log('✅ Number Plate marked Yes by ' + user.name);
          }
        }
        
        SpreadsheetApp.flush();
        
        Logger.log('=== updateOperatorStatus SUCCESS ===');
        
        return {
          success: true,
          message: 'Status updated successfully'
        };
      }
    }
    
    return {
      success: false,
      message: 'Record not found'
    };
    
  } catch (error) {
    Logger.log('❌ Error in updateOperatorStatus: ' + error.toString());
    Logger.log('Error stack: ' + error.stack);
    return {
      success: false,
      message: 'Error updating status: ' + error.toString()
    };
  }
}

        if (!pmDetails.success) {
          return {
            success: false,
            message: 'Price not found in PriceMaster for ' + model + ' ' + variant
          };
        }
        
        const pm = pmDetails.details;
        
        // Start with base prices
        let total = 0;
        total += parseFloat(pm.exShowroom) || 0;
        total += parseFloat(pm.rto) || 0;
        total += parseFloat(pm.insurance) || 0;
        total += parseFloat(pm.serviceCharge) || 0;
        total += parseFloat(pm.mandAccessories) || 0;
        
        const breakdown = {
          exShowroom: pm.exShowroom || 0,
          rto: pm.rto || 0,
          insurance: pm.insurance || 0,
          serviceCharge: pm.serviceCharge || 0,
          mandAccessories: pm.mandAccessories || 0,
          accessories: [],
          financeComm: 0
        };
        
        // Add selected accessories
        if (row[7] && pm.guardPrice) { // Guard
          const price = parseFloat(pm.guardPrice);
          total += price;
          breakdown.accessories.push({ name: 'Guard', price: price });
        }
        if (row[8] && pm.gripPrice) { // Grip
          const price = parseFloat(pm.gripPrice);
          total += price;
          breakdown.accessories.push({ name: 'Grip Cover', price: price });
        }
        if (row[9] && pm.seatCoverPrice) { // Seat Cover
          const price = parseFloat(pm.seatCoverPrice);
          total += price;
          breakdown.accessories.push({ name: 'Seat Cover', price: price });
        }
        if (row[10] && pm.matinPrice) { // Matin
          const price = parseFloat(pm.matinPrice);
          total += price;
          breakdown.accessories.push({ name: 'Matin', price: price });
        }
        if (row[11] && pm.tankCoverPrice) { // Tank Cover
          const price = parseFloat(pm.tankCoverPrice);
          total += price;
          breakdown.accessories.push({ name: 'Tank Cover', price: price });
        }
        if (row[12] && pm.handleHookPrice) { // Handle Hook
          const price = parseFloat(pm.handleHookPrice);
          total += price;
          breakdown.accessories.push({ name: 'Handle Hook', price: price });
        }
        if (row[13] && pm.helmetPrice) { // Helmet
          const helmetQty = parseInt(row[13]) || 1;
          const price = parseFloat(pm.helmetPrice) * helmetQty;
          total += price;
          breakdown.accessories.push({ name: 'Helmet (x' + helmetQty + ')', price: price });
        }
        if (row[61] === 'Yes' && pm.rainCoverPrice) {
          const price = parseFloat(pm.rainCoverPrice); total += price;
          breakdown.accessories.push({ name: 'Rain Cover', price });
        }
        if (row[62] === 'Yes' && pm.buzzerPrice) {
          const price = parseFloat(pm.buzzerPrice); total += price;
          breakdown.accessories.push({ name: 'Buzzer', price });
        }
        if (row[63] === 'Yes' && pm.backRestPrice) {
          const price = parseFloat(pm.backRestPrice); total += price;
          breakdown.accessories.push({ name: 'Back Rest', price });
        }

        
        // Add Finance Commission if not Cash
        if (financeCompany && financeCompany !== 'Cash') {
          // Check if finance comm is stored (we'll add this column)
          const financeComm = row[56] || 0; // Column BD will be moved, using temp index
          if (financeComm) {
            const commAmount = parseFloat(financeComm);
            total += commAmount;
            breakdown.financeComm = commAmount;
          }
        }
        
        return {
          success: true,
          calculation: {
            breakdown: breakdown,
            calculatedTotal: Math.round(total),
            enteredTotal: row[40] || 0 // Column AO: Total Amount (if exists)
          }
        };
      }
    }
    
    return {
      success: false,
      message: 'Record not found'
    };
    
  } catch (error) {
    Logger.log('Error in calculatePrice: ' + error.toString());
    return {
      success: false,
      message: 'Error calculating price: ' + error.toString()
    };
  }
}

/**
 * Search view records with multiple filters
 */
function searchViewRecords(sessionId, searchBy, searchValue, dateFilter, month, fromDate, toDate) {
  try {
    const validation = validateSession(sessionId);
    if (!validation.success) {
      return validation;
    }
    
    // All roles can access view records
    
    const sheet = getSheet(CONFIG.SHEETS.DATA);
    const data = sheet.getDataRange().getValues();
    
    const results = [];
    
    // Default searchValue to empty string if undefined
    searchValue = searchValue || '';
    
    // Search through data (skip header)
    for (let i = 1; i < data.length; i++) {
      const row = data[i];
      let matches = false;
      
      if (searchBy === 'Customer Name') {
        matches = row[3] && row[3].toString().toLowerCase().includes(searchValue.toLowerCase());
      } else if (searchBy === 'Mobile No') {
        matches = row[4] && row[4].toString().includes(searchValue);
      } else if (searchBy === 'Executive Name') {
        matches = row[1] && row[1].toString().toLowerCase().includes(searchValue.toLowerCase());
      } else if (searchBy === 'Booking Date') {
        const recordDate = new Date(row[2]); // Column C: Date
        
        if (dateFilter === 'date' && searchValue) {
          const targetDate = new Date(searchValue);
          matches = recordDate.toDateString() === targetDate.toDateString();
        } else if (dateFilter === 'range' && fromDate && toDate) {
          const from = new Date(fromDate);
          const to = new Date(toDate);
          matches = recordDate >= from && recordDate <= to;
        } else if (dateFilter === 'month' && month) {
          const [year, monthNum] = month.split('-');
          matches = recordDate.getFullYear() == year && (recordDate.getMonth() + 1) == monthNum;
        }
      }
      
      if (matches) {
        results.push({
          row: i + 1, // Sheet row number (1-indexed)
          receiptNo: row[0],
          date: formatDate(row[2]),
          customerName: row[3],
          mobileNo: row[4],
          model: row[5],
          variant: row[6]
        });
      }
    }
    
    return {
      success: true,
      results: results,
      message: 'Found ' + results.length + ' record(s)'
    };
    
  } catch (error) {
    Logger.log('Error in searchViewRecords: ' + error.toString());
    return {
      success: false,
      message: 'Error searching records: ' + error.toString()
    };
  }
}

/**
 * Get accessory dashboard data
 */
function getAccessoryDashboardData(sessionId, month) {
  try {
    const validation = validateSession(sessionId);
    if (!validation.success) {
      return validation;
    }
    
    const user = validation.user;
    if (user.role !== 'admin' && user.role !== 'accessories') {
      return { success: false, message: 'Access denied' };
    }
    
    const sheet = getSheet(CONFIG.SHEETS.DATA);
    const data = sheet.getDataRange().getValues();
    
    // Parse month (format: YYYY-MM)
    const [year, monthNum] = month.split('-');

    let yesCount = 0;
    let accountsYesAccessoryBlankCount = 0;
    let blankCount = 0;
    let partialCount = 0;
    let todayDeliveryCount = 0;

    // Today's date (midnight) for same-day comparison
    const todayStart = new Date();
    todayStart.setHours(0, 0, 0, 0);
    const todayEnd = new Date();
    todayEnd.setHours(23, 59, 59, 999);

    // Single pass: count monthly stats AND today's deliveries together
    for (let i = 1; i < data.length; i++) {
      const row = data[i];
      const deliveryDate = row[11]; // Column L: Delivery Date
      const accountCheck = row[21]; // Column V: Account Check
      const accessoryFitted = row[36]; // Column AK: Accessory Fitted

      if (!deliveryDate) continue;
      const recordDate = new Date(deliveryDate);

      // Today's delivery count (all statuses)
      if (recordDate >= todayStart && recordDate <= todayEnd) {
        todayDeliveryCount++;
      }

      // Monthly stats
      if (recordDate.getFullYear() == year && (recordDate.getMonth() + 1) == monthNum) {
        if (accessoryFitted === 'Yes') {
          yesCount++;
        } else if (accountCheck === 'Yes' && (!accessoryFitted || accessoryFitted === '')) {
          accountsYesAccessoryBlankCount++;
        } else if (!accessoryFitted || accessoryFitted === '') {
          blankCount++;
        } else if (accessoryFitted === 'No') {
          partialCount++;
        }
      }
    }

    return {
      success: true,
      data: {
        yes: yesCount,
        accountsYesAccessoryBlank: accountsYesAccessoryBlankCount,
        blank: blankCount,
        partial: partialCount,
        total: yesCount + accountsYesAccessoryBlankCount + blankCount + partialCount,
        todayDelivery: todayDeliveryCount
      }
    };
    
  } catch (error) {
    Logger.log('Error in getAccessoryDashboardData: ' + error.toString());
    return {
      success: false,
      message: 'Error loading dashboard: ' + error.toString()
    };
  }
}

/**
 * Calculate monthly trend for last 6 months (completed sales only)
 */
function calculateMonthlyTrend(data, executiveName) {
  const now = new Date();
  const months = [];
  
  // Generate last 6 months
  for (let i = 5; i >= 0; i--) {
    const d = new Date(now.getFullYear(), now.getMonth() - i, 1);
    months.push({
      year: d.getFullYear(),
      month: d.getMonth(),
      label: d.toLocaleDateString('en-US', { month: 'short', year: 'numeric' }),
      count: 0
    });
  }
  
  // Count completed sales per month
  for (let i = 1; i < data.length; i++) {
    const row = data[i];
    const executive = row[1];
    const bookingDate = row[2];
    const accountCheck = row[21];
    
    if (executive === executiveName && accountCheck === 'Yes' && bookingDate) {
      const date = new Date(bookingDate);
      
      for (let j = 0; j < months.length; j++) {
        if (date.getFullYear() === months[j].year && date.getMonth() === months[j].month) {
          months[j].count++;
          break;
        }
      }
    }
  }
  
  // Calculate percentages (relative to max)
  const maxCount = Math.max.apply(null, months.map(function(m) { return m.count; }).concat([1]));
  
  return months.map(function(m) {
    return {
      month: m.label,
      count: m.count,
      percentage: Math.round((m.count / maxCount) * 100)
    };
  });
}

      if (!receiptNo) {

      }
      
      // Filter conditions:

      const isUserRecord = (user.role === 'admin' || user.role === 'accounts') ||

      const engineFilled = engineNumber && String(engineNumber).trim() !== '';
      const frameFilled = frameNumber && String(frameNumber).trim() !== '';
      const isComplete = engineFilled && frameFilled;

/**
 * Parse CSV and check for conflicts
 * Replaces the existing parseAndCheckExcel function
 */
function parseAndCheckExcel(sessionId, base64Data, fileName) {
  try {
    const validation = validateSession(sessionId);
    if (!validation.success) return validation;

    const user = validation.user;
    if (user.role !== 'operator' && user.role !== 'admin') {
      return { success: false, message: 'Access denied' };
    }

    // ── Decode base64 CSV ──────────────────────────────────────────────────
    const decoded = Utilities.newBlob(
      Utilities.base64Decode(base64Data),
      'text/csv'
    ).getDataAsString('UTF-8');

    // ── Parse CSV rows ─────────────────────────────────────────────────────
    const rows = decoded
      .replace(/\r\n/g, '\n')
      .replace(/\r/g, '\n')
      .split('\n')
      .filter(function(line) { return line.trim() !== ''; });

    if (rows.length < 2) {
      return { success: false, message: 'CSV file is empty or has no data rows' };
    }

    Logger.log('=== parseAndCheckExcel (HSRP CSV) ===');
    Logger.log('Total CSV rows (including header): ' + rows.length);

    // ── Load Data sheet ────────────────────────────────────────────────────
    const sheet     = getSheet(CONFIG.SHEETS.DATA);
    const sheetData = sheet.getDataRange().getValues();

    // Build a lookup map: frameNumber (BI = index 60) → row index
    // Also store current registration number (Col AZ = index 51) for conflict check
    const frameMap = {};  // { frameNumber: { rowIndex, currentReg } }

    for (var r = 1; r < sheetData.length; r++) {
      const frameVal = String(sheetData[r][60] || '').trim().toUpperCase();
      if (frameVal) {
        frameMap[frameVal] = {
          rowIndex:   r,
          currentReg: String(sheetData[r][51] || '').trim()
        };
      }
    }

    Logger.log('Data sheet rows with frame numbers: ' + Object.keys(frameMap).length);

    // ── Process CSV rows (skip header row 0) ──────────────────────────────
    const validRecords   = [];
    const conflicts      = [];
    const skippedDetails = [];

    for (var i = 1; i < rows.length; i++) {
      const cols = parseCSVLine(rows[i]);

      // CSV: A=0 RegDate, B=1 RegNumber, C=2 CustomerName, D=3 FrameNo
      const regDate   = String(cols[0] || '').trim();
      const regNumber = String(cols[1] || '').trim().toUpperCase().replace(/\s+/g, '');
      const custName  = String(cols[2] || '').trim();
      const frameNo   = String(cols[3] || '').trim().toUpperCase();

      // Skip rows missing frame number or registration number
      if (!frameNo) {
        skippedDetails.push({ row: i + 1, reason: 'Missing Frame Number (Column D)' });
        continue;
      }
      if (!regNumber) {
        skippedDetails.push({ row: i + 1, reason: 'Missing Registration Number (Column B)' });
        continue;
      }

      const match = frameMap[frameNo];

      if (!match) {
        // Frame number not found in Data sheet — will be reported as "not found"
        validRecords.push({
          frameNumber:          frameNo,
          registrationNumber:   regNumber,
          registrationDate:     regDate,
          customerName:         custName,
          foundInSheet:         false
        });
        continue;
      }

      // Check conflict: Data sheet already has a different registration number
      if (match.currentReg && match.currentReg !== regNumber) {
        conflicts.push({
          chassisNumber: frameNo,
          currentPlate:  match.currentReg,
          newPlate:      regNumber
        });
      }

      validRecords.push({
        frameNumber:          frameNo,
        registrationNumber:   regNumber,
        registrationDate:     regDate,
        customerName:         custName,
        foundInSheet:         true,
        rowIndex:             match.rowIndex
      });
    }

    Logger.log('Valid records: ' + validRecords.length);
    Logger.log('Conflicts: ' + conflicts.length);
    Logger.log('Skipped: ' + skippedDetails.length);

    return {
      success:         true,
      validRecords:    validRecords,
      conflicts:       conflicts,
      skippedDetails:  skippedDetails,
      totalParsed:     rows.length - 1
    };

  } catch (e) {
    Logger.log('parseAndCheckExcel error: ' + e.toString());
    return { success: false, message: 'Error parsing CSV: ' + e.toString() };
  }
}

/**
 * Parse a single CSV line, handling quoted values
 */
function parseCSVLine(line) {
  const values = [];
  let current = '';
  let inQuotes = false;
  
  for (let i = 0; i < line.length; i++) {
    const char = line[i];
    
    if (char === '"') {
      inQuotes = !inQuotes;
    } else if (char === ',' && !inQuotes) {
      values.push(current);
      current = '';
    } else {
      current += char;
    }
  }
  
  // Push last value
  values.push(current);
  
  return values;
}

            });
          }
          break; // Found match, stop searching
        }
      }
    }
    
    Logger.log('Found ' + conflicts.length + ' conflicts');
    
    return {
      success: true,
      conflicts: conflicts
    };

    };

/**
 * Bulk update registration numbers in Data sheet
 * Matches on Frame Number (Col BI, index 60) and writes to Col AZ (index 51)
 * Replaces the existing bulkUpdateNumberPlates function
 */
function bulkUpdateNumberPlates(sessionId, records, overwriteExisting) {
  try {
    const validation = validateSession(sessionId);
    if (!validation.success) return validation;

    const user = validation.user;
    if (user.role !== 'operator' && user.role !== 'admin') {
      return { success: false, message: 'Access denied' };
    }

    if (!records || records.length === 0) {
      return { success: false, message: 'No records to update' };
    }

    Logger.log('=== bulkUpdateNumberPlates ===');
    Logger.log('Records received: ' + records.length);
    Logger.log('Overwrite existing: ' + overwriteExisting);

    const sheet     = getSheet(CONFIG.SHEETS.DATA);
    const sheetData = sheet.getDataRange().getValues();

    // Rebuild frame→row map fresh (in case sheet changed since parse step)
    const frameMap = {};
    for (var r = 1; r < sheetData.length; r++) {
      const frameVal = String(sheetData[r][60] || '').trim().toUpperCase();
      if (frameVal) {
        frameMap[frameVal] = r;
      }
    }

    var updated        = 0;
    var skipped        = 0;
    var notFound       = 0;
    const details          = [];
    const skippedDetails   = [];
    const notFoundDetails  = [];

    for (var i = 0; i < records.length; i++) {
      const rec        = records[i];
      const frameNo    = String(rec.frameNumber   || '').trim().toUpperCase();
      const regNumber  = String(rec.registrationNumber || '').trim().toUpperCase().replace(/\s+/g, '');

      if (!frameNo || !regNumber) {
        skipped++;
        skippedDetails.push({ row: i + 1, reason: 'Missing frame number or registration number' });
        continue;
      }

      const rowIndex = frameMap[frameNo];

      if (rowIndex === undefined) {
        notFound++;
        notFoundDetails.push({
          chassisNumber:       frameNo,
          registrationNumber:  regNumber
        });
        continue;
      }

      // Check if already has a registration number (Col AZ = index 51)
      const existingReg = String(sheetData[rowIndex][51] || '').trim();

      if (existingReg && existingReg !== regNumber && !overwriteExisting) {
        skipped++;
        skippedDetails.push({
          row:    rowIndex + 1,
          reason: 'Already has registration number: ' + existingReg + ' (overwrite not confirmed)'
        });
        continue;
      }

      // ── Write registration number to Col AZ (index 51 → sheet col 52) ──
      sheet.getRange(rowIndex + 1, 52).setValue(regNumber);

      updated++;
      details.push({
        chassisNumber:      frameNo,
        registrationNumber: regNumber,
        receiptNo:          rec.registrationDate || ''
      });

      Logger.log('Updated row ' + (rowIndex + 1) + ': Frame=' + frameNo + ' → Reg=' + regNumber);
    }

    Logger.log('Done: updated=' + updated + ', skipped=' + skipped + ', notFound=' + notFound);

    return {
      success: true,
      results: {
        total:           records.length,
        updated:         updated,
        skipped:         skipped,
        notFound:        notFound,
        details:         details,
        skippedDetails:  skippedDetails,
        notFoundDetails: notFoundDetails
      }
    };

  } catch (e) {
    Logger.log('bulkUpdateNumberPlates error: ' + e.toString());
    return { success: false, message: 'Error updating records: ' + e.toString() };
  }
}

/**
 * Upload V301 file (Step 1)
 * Processes invoice CSV and imports to HSRP sheet
 */
function uploadV301File(sessionId, base64Data, fileName) {
  try {
    const validation = validateSession(sessionId);
    if (!validation.success) {
      return validation;
    }
    
    const user = validation.user;
    
    // Only operator and admin can access
    if (user.role !== 'operator' && user.role !== 'admin') {
      return {
        success: false,
        message: 'Access denied'
      };
    }
    
    Logger.log('=== uploadV301File ===');
    Logger.log('User: ' + user.name);
    Logger.log('File: ' + fileName);
    
    // Decode base64 to CSV text
    const csvText = Utilities.newBlob(Utilities.base64Decode(base64Data)).getDataAsString();
    
    // Parse CSV
    const lines = csvText.split(/\r?\n/);
    
    Logger.log('Total lines: ' + lines.length);
    
    // Delete rows 1, 2, and 4 (indices 0, 1, 3)
    // Keep row 3 (index 2) as header
    const dataLines = [];
    for (let i = 0; i < lines.length; i++) {
      if (i === 0 || i === 1 || i === 3) {
        // Skip these rows
        continue;
      }
      dataLines.push(lines[i]);
    }
    
    Logger.log('After deleting rows 1,2,4: ' + dataLines.length + ' lines');
    
    if (dataLines.length < 2) {
      return {
        success: false,
        message: 'No data found in CSV file'
      };
    }
    
    // Get HSRP sheet
    const hsrpSheet = getSheet('HSRP');
    const existingData = hsrpSheet.getDataRange().getValues();
    
    // Build set of existing invoice numbers for duplicate check
    const existingInvoices = {};
    for (let i = 1; i < existingData.length; i++) {
      const invoiceNo = existingData[i][1]; // Column B
      if (invoiceNo) {
        existingInvoices[invoiceNo.toString()] = true;
      }
    }
    
    // Get current max Sr No
    let maxSrNo = 0;
    for (let i = 1; i < existingData.length; i++) {
      const srNo = parseInt(existingData[i][0]) || 0;
      if (srNo > maxSrNo) {
        maxSrNo = srNo;
      }
    }
    
    Logger.log('Max existing Sr No: ' + maxSrNo);
    
    // Parse data rows (skip header at index 0)
    const rowsToAdd = [];
    const skippedRows = [];
    const today = new Date();
    
    for (let i = 1; i < dataLines.length; i++) {
      const line = dataLines[i].trim();
      
      if (!line) continue;
      
      const values = parseCSVLine(line);
      
      // Extract columns (0-indexed)
      const invoiceDate = values[1] || '';    // Column B
      const invoiceNo = values[2] || '';      // Column C
      const modelName = values[3] || '';      // Column D
      const frameNo = values[4] || '';        // Column E
      const customerName = values[6] || '';   // Column G
      const mobileNo = values[8] || '';       // Column I
      const hpCompany = values[9] || '';      // Column J - HP Company
      const refCustomer = values[12] || '';   // Column M
      
      // Skip if invoice number already exists
      if (existingInvoices[invoiceNo.toString()]) {
        skippedRows.push({
          invoiceNo: invoiceNo,
          customerName: customerName
        });
        continue;
      }
      
      // Increment Sr No
      maxSrNo++;
      
      // Build row for HSRP sheet
      // Columns: Sr No, Invoice No, Invoice Date, Status, Mobile No, Customer Name, Frame No, Registration No, Ref Customer, Model Name, Order Date, Fitment Date, HP Company
      const newRow = [
          maxSrNo,                  // A: Sr No
          invoiceNo,                // B: Invoice No
          invoiceDate,              // C: Invoice Date
          '',                       // D: Status (blank initially)
          mobileNo,                 // E: Mobile No
          customerName,             // F: Customer Name
          frameNo,                  // G: Frame No
          '',                       // H: Registration No (blank)
          (customerName.toUpperCase().includes('SEVA MOTORS') ? 'SEVA MOTORS' : (refCustomer || 'Vinay Automobiles')),  // I: Ref Customer
          modelName,                // J: Model Name
          '',                       // K: Order Date (blank initially, set during Step 2)
          '',                       // L: Fitment Date (blank)
          hpCompany                 // M: HP Company (stored, not displayed)
      ];
      
      rowsToAdd.push(newRow);
      
      // Mark as existing to avoid duplicates within same upload
      existingInvoices[invoiceNo.toString()] = true;
    }
    
    Logger.log('Rows to add: ' + rowsToAdd.length);
    Logger.log('Skipped (duplicates): ' + skippedRows.length);
    
    // Append rows to sheet
    if (rowsToAdd.length > 0) {
      const startRow = hsrpSheet.getLastRow() + 1;
      hsrpSheet.getRange(startRow, 1, rowsToAdd.length, 13).setValues(rowsToAdd);
      SpreadsheetApp.flush();
    }
    
    return {
      success: true,
      message: 'Invoice data imported successfully',
      results: {
        total: dataLines.length - 1, // Exclude header
        imported: rowsToAdd.length,
        skipped: skippedRows.length,
        skippedDetails: skippedRows
      }
    };
    
  } catch (error) {
    Logger.log('Error in uploadV301File: ' + error.toString());
    Logger.log('Error stack: ' + error.stack);
    return {
      success: false,
      message: 'Error processing file: ' + error.toString()
    };
  }
}

/**
 * Helper: Parse a single CSV line (handles quoted fields with commas inside)
 * Add this to Code.gs if not already present
 */
function parseCSVLine(line) {
  const result = [];
  let current  = '';
  let inQuotes = false;

  for (var c = 0; c < line.length; c++) {
    const ch = line[c];
    if (ch === '"') {
      inQuotes = !inQuotes;
    } else if (ch === ',' && !inQuotes) {
      result.push(current.trim());
      current = '';
    } else {
      current += ch;
    }
  }
  result.push(current.trim());
  return result;
}

/**
 * Get HSRP data for viewing
 * UPDATED: Format Invoice Date (date only), Order Date (date + time IST)
 */
function getHSRPData(sessionId) {
  try {
    const validation = validateSession(sessionId);
    if (!validation.success) {
      return validation;
    }
    
    const user = validation.user;
    
    // Only operator and admin can access
    if (user.role !== 'operator' && user.role !== 'admin') {
      return {
        success: false,
        message: 'Access denied'
      };
    }
    
    const sheet = getSheet('HSRP');
    const data = sheet.getDataRange().getValues();
    
    const records = [];
    
    // Helper function to format date only (no time)
    function formatDateOnly(dateValue) {
      if (!dateValue) return '';
      try {
        if (typeof dateValue === 'string') {
          // Already a string, remove time if present
          return dateValue.split(' ')[0];
        }
        const d = new Date(dateValue);
        return Utilities.formatDate(d, 'Asia/Kolkata', 'dd-MMM-yyyy');
      } catch (e) {
        return dateValue.toString().split(' ')[0];
      }
    }
    
    // Helper function to format date with time (IST)
    function formatDateTimeIST(dateValue) {
      if (!dateValue) return '';
      try {
        if (typeof dateValue === 'string') return dateValue;
        const d = new Date(dateValue);
        return Utilities.formatDate(d, 'Asia/Kolkata', 'dd-MMM-yyyy HH:mm:ss');
      } catch (e) {
        return dateValue.toString();
      }
    }
    
    // Skip header row
    for (let i = 1; i < data.length; i++) {
      const row = data[i];
      
      records.push({
        srNo: row[0],
        invoiceNo: row[1],
        invoiceDate: formatDateOnly(row[2]),        // Date only, no time
        status: row[3],
        mobileNo: row[4],
        customerName: row[5],
        frameNo: row[6],
        registrationNo: row[7],
        refCustomer: row[8],
        modelName: row[9],
        orderDate: formatDateTimeIST(row[10]),      // Date + Time (IST)
        fitmentDate: formatDateOnly(row[11]),       // Date only, no time
        hpCompany: row[12] || ''                   // Stored but not displayed
      });
    }
    
    return {
      success: true,
      data: records
    };
    
  } catch (error) {
    Logger.log('Error in getHSRPData: ' + error.toString());
    return {
      success: false,
      message: 'Error loading data: ' + error.toString()
    };
  }
}

/**
 * Download HSRP data as CSV
 */
function downloadHSRPData(sessionId) {
  try {
    const validation = validateSession(sessionId);
    if (!validation.success) {
      return validation;
    }
    
    const user = validation.user;
    
    // ONLY ADMIN can download CSV
    if (user.role !== 'admin') {
      return {
        success: false,
        message: 'Access denied. Only Admin can download CSV.'
      };
    }
    
    const sheet = getSheet('HSRP');
    const data = sheet.getDataRange().getValues();
    
    // Build CSV
    let csv = '';
    
    for (let i = 0; i < data.length; i++) {
      const row = data[i];
      
      const csvRow = row.map(function(cell) {
        return '"' + (cell || '').toString().replace(/"/g, '""') + '"';
      }).join(',');
      
      csv += csvRow + '\n';
    }
    
    return {
      success: true,
      csv: csv
    };
    
  } catch (error) {
    Logger.log('Error in downloadHSRPData: ' + error.toString());
    return {
      success: false,
      message: 'Error downloading data: ' + error.toString()
    };
  }
}

/**
 * Search HSRP data
 * Search by: Invoice No, Customer Name, Registration No, Invoice Date
 */
function searchHSRPData(sessionId, searchBy, searchValue, dateFilter, customDate) {
  try {
    const validation = validateSession(sessionId);
    if (!validation.success) {
      return validation;
    }
    
    const user = validation.user;
    
    // Only operator and admin can access
    if (user.role !== 'operator' && user.role !== 'admin') {
      return {
        success: false,
        message: 'Access denied'
      };
    }
    
    const sheet = getSheet('HSRP');
    const data = sheet.getDataRange().getValues();
    
    const results = [];
    
    // Helper function to format date only (no time)
    function formatDateOnly(dateValue) {
      if (!dateValue) return '';
      try {
        if (typeof dateValue === 'string') {
          return dateValue.split(' ')[0];
        }
        const d = new Date(dateValue);
        return Utilities.formatDate(d, 'Asia/Kolkata', 'dd-MMM-yyyy');
      } catch (e) {
        return dateValue.toString().split(' ')[0];
      }
    }
    
    // Helper function to format date with time (IST)
    function formatDateTimeIST(dateValue) {
      if (!dateValue) return '';
      try {
        if (typeof dateValue === 'string') return dateValue;
        const d = new Date(dateValue);
        return Utilities.formatDate(d, 'Asia/Kolkata', 'dd-MMM-yyyy HH:mm:ss');
      } catch (e) {
        return dateValue.toString();
      }
    }
    
    const today = new Date();
    today.setHours(0, 0, 0, 0);
    
    // Search through data (skip header)
    for (let i = 1; i < data.length; i++) {
      const row = data[i];
      let matches = false;
      
      if (searchBy === 'invoiceNo') {
        matches = row[1] && row[1].toString().toLowerCase().includes(searchValue.toLowerCase());
      } else if (searchBy === 'customerName') {
        matches = row[5] && row[5].toString().toLowerCase().includes(searchValue.toLowerCase());
      } else if (searchBy === 'registrationNo') {
        matches = row[7] && row[7].toString().toLowerCase().includes(searchValue.toLowerCase());
      } else if (searchBy === 'refCustomer') {  
     // NEW: Search by Ref Customer (Column I, index 8)
      matches = row[8] && row[8].toString().toLowerCase().includes(searchValue.toLowerCase());
    } else if (searchBy === 'orderDate') {
    // NEW: Search by Order Date (Column K, index 10)
    const orderDate = row[10];
  if (orderDate) {
    try {
      const orderDateObj = new Date(orderDate);
      orderDateObj.setHours(0, 0, 0, 0);
      
      if (dateFilter === 'today') {
        const today = new Date();
        today.setHours(0, 0, 0, 0);
        matches = orderDateObj.getTime() === today.getTime();
      } else if (dateFilter === 'customDate' && customDate) {
        const targetDate = new Date(customDate);
        targetDate.setHours(0, 0, 0, 0);
        matches = orderDateObj.getTime() === targetDate.getTime();
      }
    } catch (e) {
      Logger.log('Error comparing order date: ' + e.toString());
    }
  }

      
      } else if (searchBy === 'invoiceDate') {
        const invoiceDate = row[2]; // Column C: Invoice Date
        
        if (invoiceDate) {
          const recordDate = new Date(invoiceDate);
          recordDate.setHours(0, 0, 0, 0);
          
          if (dateFilter === 'today') {
            matches = recordDate.getTime() === today.getTime();
          } else if (dateFilter === 'thisMonth') {
            matches = recordDate.getFullYear() === today.getFullYear() && 
                     recordDate.getMonth() === today.getMonth();
          } else if (dateFilter === 'customDate' && customDate) {
            const targetDate = new Date(customDate);
            targetDate.setHours(0, 0, 0, 0);
            matches = recordDate.getTime() === targetDate.getTime();
          }
        }
      }
      
      if (matches) {
        results.push({
          srNo: row[0],
          invoiceNo: row[1],
          invoiceDate: formatDateOnly(row[2]),
          status: row[3],
          mobileNo: row[4],
          customerName: row[5],
          frameNo: row[6],
          registrationNo: row[7],
          refCustomer: row[8],
          modelName: row[9],
          orderDate: formatDateTimeIST(row[10]),
          fitmentDate: formatDateOnly(row[11]),
          hpCompany: row[12] || ''             // Stored but not displayed
        });
      }
    }
    
    return {
      success: true,
      data: results,
      count: results.length
    };
    
  } catch (error) {
    Logger.log('Error in searchHSRPData: ' + error.toString());
    return {
      success: false,
      message: 'Error searching data: ' + error.toString()
    };
  }
}

/**
 * Parse a CSV line (handles quoted values)
 * Helper function
 */
function parseCSVLine(line) {
  const values = [];
  let current = '';
  let inQuotes = false;
  
  for (let i = 0; i < line.length; i++) {
    const char = line[i];
    
    if (char === '"') {
      inQuotes = !inQuotes;
    } else if (char === ',' && !inQuotes) {
      values.push(current);
      current = '';
    } else {
      current += char;
    }
  }
  
  // Push last value
  values.push(current);
  
  return values;
}


/**
 * Update HSRP status (inline editing from view data table)
 * When status = "Fitted", also set fitment date to today
 * When order date exists in Col K, ensure status is set to "Ordered"
 */
function updateHSRPStatus(sessionId, srNo, newStatus) {
  try {
    const validation = validateSession(sessionId);
    if (!validation.success) return validation;

    const user = validation.user;
    if (user.role !== 'operator' && user.role !== 'admin') {
      return { success: false, message: 'Access denied' };
    }

    Logger.log('=== updateHSRPStatus ===');
    Logger.log('Sr No: ' + srNo + ' → Status: ' + newStatus);

    const hsrpSheet = getSheet('HSRP');
    const hsrpData  = hsrpSheet.getDataRange().getValues();

    // Find row by Sr No (Column A, index 0)
    var targetRow = -1;
    for (var i = 1; i < hsrpData.length; i++) {
      if (hsrpData[i][0] == srNo) {
        targetRow = i;
        break;
      }
    }

    if (targetRow === -1) {
      return { success: false, message: 'Record not found with Sr No: ' + srNo };
    }

    const sheetRow = targetRow + 1; // Convert to 1-based sheet row

    // Update Status in Column D (index 3 → sheet column 4)
    hsrpSheet.getRange(sheetRow, 4).setValue(newStatus);

    // If status = "Fitted", set Fitment Date to today in Column L (index 11 → sheet column 12)
    if (newStatus === 'Fitted') {
      const today = new Date();
      const fitmentDate = Utilities.formatDate(today, 'Asia/Kolkata', 'dd-MMM-yyyy');
      hsrpSheet.getRange(sheetRow, 12).setValue(fitmentDate);
      Logger.log('✅ Set Fitment Date: ' + fitmentDate);
    }

    SpreadsheetApp.flush();

    Logger.log('✅ Status updated successfully');

    return {
      success: true,
      message: 'Status updated to: ' + newStatus
    };

  } catch (error) {
    Logger.log('Error in updateHSRPStatus: ' + error.toString());
    return {
      success: false,
      message: 'Error updating status: ' + error.toString()
    };
  }
}


/**
 * Export HSRP to PDF with date range filter
 * Returns data to frontend for PDF generation
 * UPDATED: Now includes mobileNo and modelName
 */
function exportHSRPToPdf(sessionId, fromDate, toDate) {
  try {
    const validation = validateSession(sessionId);
    if (!validation.success) {
      return validation;
    }
    
    const user = validation.user;
    
    // Only operator and admin can export
    if (user.role !== 'operator' && user.role !== 'admin') {
      return {
        success: false,
        message: 'Access denied'
      };
    }
    
    Logger.log('=== exportHSRPToPdf ===');
    Logger.log('User: ' + user.name);
    Logger.log('Date Range: ' + fromDate + ' to ' + toDate);
    
    // Parse dates
    const fromDateObj = new Date(fromDate);
    const toDateObj = new Date(toDate);
    toDateObj.setHours(23, 59, 59, 999); // Include entire end date
    
    // Get HSRP data
    const hsrpSheet = getSheet('HSRP');
    const data = hsrpSheet.getDataRange().getValues();
    
    // Filter by invoice date range
    const filteredRecords = [];
    
    for (let i = 1; i < data.length; i++) {
      const row = data[i];
      const invoiceDate = row[2]; // Column C: Invoice Date
      
      if (invoiceDate) {
        const invoiceDateObj = new Date(invoiceDate);
        
        if (invoiceDateObj >= fromDateObj && invoiceDateObj <= toDateObj) {
          filteredRecords.push({
            invoiceNo: row[1] || '',
            invoiceDate: formatDateOnly(invoiceDate),
            customerName: row[5] || '',
            mobileNo: row[4] || '',         // Column E: Mobile No
            modelName: row[9] || '',        // Column J: Model Name
            frameNo: row[6] || '',
            plateNo: row[7] || ''           // Column H: Registration No
          });
        }
      }
    }
    
    Logger.log('Filtered records: ' + filteredRecords.length);
    
    if (filteredRecords.length === 0) {
      return {
        success: false,
        message: 'No records found in the selected date range'
      };
    }
    
    // Return records to frontend for PDF generation
    return {
      success: true,
      message: 'Data ready for PDF export',
      records: filteredRecords,
      fromDate: fromDate,
      toDate: toDate,
      recordCount: filteredRecords.length
    };
    
  } catch (error) {
    Logger.log('Error in exportHSRPToPdf: ' + error.toString());
    return {
      success: false,
      message: 'Error generating PDF: ' + error.toString()
    };
  }
}

/**
 * Get HSRP records with financier data
 * Filters: customerName, refCustomer, financier (HP Company) — all optional partial match
 */
function getFinancierData(params, sessionId) {
  try {
    const validation = validateSession(sessionId);
    if (!validation.success) return { success: false, message: validation.message || 'Session expired' };
    const user = validation.user;
    if (user.role !== 'admin' && user.role !== 'accounts') {
      return { success: false, message: 'Access denied' };
    }

    const customerName = (params.customerName || '').trim().toLowerCase();
    const refCustomer  = (params.refCustomer  || '').trim().toLowerCase();
    const financier    = (params.financier    || '').trim().toLowerCase();

    const sheet = getSheet('HSRP');
    const data = sheet.getDataRange().getValues();

    function fmtDateDisplay(val) {
      if (!val) return '';
      try {
        if (val instanceof Date) return Utilities.formatDate(val, 'Asia/Kolkata', 'dd-MMM-yyyy');
        return String(val).split(' ')[0];
      } catch(e) { return String(val); }
    }

    // For date inputs (receivedDate, doDate) — return in yyyy-MM-dd format
    function fmtDateInput(val) {
      if (!val) return '';
      try {
        if (val instanceof Date) return Utilities.formatDate(val, 'Asia/Kolkata', 'yyyy-MM-dd');
        const s = String(val).trim();
        // If already yyyy-MM-dd, return as-is
        if (/^\d{4}-\d{2}-\d{2}/.test(s)) return s.substring(0, 10);
        return s;
      } catch(e) { return String(val); }
    }

    const records = [];
    for (let i = 1; i < data.length; i++) {
      const row = data[i];
      if (!row[1]) continue; // skip rows without invoice no

      // Apply optional filters (partial match, case-insensitive)
      if (customerName && String(row[5]).toLowerCase().indexOf(customerName) === -1) continue;
      if (refCustomer  && String(row[8]).toLowerCase().indexOf(refCustomer)  === -1) continue;
      if (financier    && String(row[12]).toLowerCase().indexOf(financier)   === -1) continue;

      records.push({
        srNo:            row[0],
        invoiceNo:       row[1],
        invoiceDate:     fmtDateDisplay(row[2]),
        customerName:    row[5] || '',
        mobileNo:        row[4] || '',
        modelName:       row[9] || '',
        refCustomer:     row[8] || '',
        financier:       row[12] || '',      // HP Company
        disbursalAmount: row[13] !== '' && row[13] !== null && row[13] !== undefined ? row[13] : '',
        amountReceived:  row[14] !== '' && row[14] !== null && row[14] !== undefined ? row[14] : '',
        receivedDate:    fmtDateInput(row[15]),
        doDate:          fmtDateInput(row[16]),
        doNumber:        row[17] || ''
      });
    }

    return { success: true, data: records };
  } catch(e) { return { success: false, message: e.message }; }
}

/**
 * Save financier data for an invoice (columns N-R of HSRP sheet)
 */
function saveFinancierData(params, sessionId) {
  try {
    const validation = validateSession(sessionId);
    if (!validation.success) return { success: false, message: validation.message || 'Session expired' };
    const user = validation.user;
    if (user.role !== 'admin' && user.role !== 'accounts') {
      return { success: false, message: 'Access denied' };
    }

    const { invoiceNo, disbursalAmount, amountReceived, receivedDate, doDate, doNumber } = params;
    if (!invoiceNo) return { success: false, message: 'Invoice No is required' };

    const sheet = getSheet('HSRP');
    const data = sheet.getDataRange().getValues();

    let targetRow = -1;
    for (let i = 1; i < data.length; i++) {
      if (String(data[i][1]) === String(invoiceNo)) {
        targetRow = i + 1; // 1-indexed sheet row
        break;
      }
    }

    if (targetRow === -1) return { success: false, message: 'Invoice not found: ' + invoiceNo };

    // N=14, O=15, P=16, Q=17, R=18 (1-indexed columns)
    sheet.getRange(targetRow, 14).setValue(disbursalAmount !== '' && disbursalAmount !== null && disbursalAmount !== undefined ? disbursalAmount : '');
    sheet.getRange(targetRow, 15).setValue(amountReceived  !== '' && amountReceived  !== null && amountReceived  !== undefined ? amountReceived  : '');
    sheet.getRange(targetRow, 16).setValue(receivedDate || '');
    sheet.getRange(targetRow, 17).setValue(doDate       || '');
    sheet.getRange(targetRow, 18).setValue(doNumber     || '');
    SpreadsheetApp.flush();

    return { success: true, message: 'Saved successfully' };
  } catch(e) { return { success: false, message: e.message }; }
}

/**
 * Batch save financier data for multiple invoices
 */
function batchSaveFinancierData(params, sessionId) {
  try {
    const validation = validateSession(sessionId);
    if (!validation.success) return { success: false, message: validation.message || 'Session expired' };
    const user = validation.user;
    if (user.role !== 'admin' && user.role !== 'accounts') {
      return { success: false, message: 'Access denied' };
    }

    const recs = params.records || [];
    if (recs.length === 0) return { success: false, message: 'No records to save' };

    const sheet = getSheet('HSRP');
    const data = sheet.getDataRange().getValues();

    // Build invoice → sheet row map once
    const invoiceRowMap = {};
    for (let i = 1; i < data.length; i++) {
      if (data[i][1]) invoiceRowMap[String(data[i][1])] = i + 1;
    }

    let saved = 0;
    const notFound = [];

    for (let k = 0; k < recs.length; k++) {
      const rec = recs[k];
      const targetRow = invoiceRowMap[String(rec.invoiceNo)];
      if (!targetRow) { notFound.push(rec.invoiceNo); continue; }

      sheet.getRange(targetRow, 14).setValue(rec.disbursalAmount !== '' && rec.disbursalAmount !== null ? rec.disbursalAmount : '');
      sheet.getRange(targetRow, 15).setValue(rec.amountReceived  !== '' && rec.amountReceived  !== null ? rec.amountReceived  : '');
      sheet.getRange(targetRow, 16).setValue(rec.receivedDate || '');
      sheet.getRange(targetRow, 17).setValue(rec.doDate       || '');
      sheet.getRange(targetRow, 18).setValue(rec.doNumber     || '');
      saved++;
    }

    SpreadsheetApp.flush();
    const msg = 'Saved ' + saved + ' record' + (saved !== 1 ? 's' : '') +
                (notFound.length ? '. Not found: ' + notFound.join(', ') : '');
    return { success: true, message: msg };
  } catch(e) { return { success: false, message: e.message }; }
}

/**
 * Save a standalone payout entry to financierpayout sheet
 * Columns: PayoutId | CreatedDate | InvoiceAmount | PayoutAmount | PayoutMonth | Notes | EnteredBy
 */
function savePayout(params, sessionId) {
  try {
    const validation = validateSession(sessionId);

      hpCompany     || ''
    ]);

    return { success: true, message: 'Payout saved successfully' };
  } catch(e) { return { success: false, message: e.message }; }

/**
 * Fetch payout entries, optionally filtered by month (yyyy-MM)
 */
function getPayouts(params, sessionId) {
  try {
    const validation = validateSession(sessionId);
    if (!validation.success) return { success: false, message: validation.message || 'Session expired' };
    const user = validation.user;
    if (user.role !== 'admin' && user.role !== 'accounts') return { success: false, message: 'Access denied' };

    const month = (params.month || '').toString().trim();

    const sheet = getSheet('financierpayout');
    const data  = sheet.getDataRange().getValues();

    const payouts = [];
    for (let i = 0; i < data.length; i++) {
      const row = data[i];
      if (!row[0]) continue;
      // Skip header
      if (String(row[0]).toLowerCase() === 'payoutid') continue;

      // Month filter
      const rowMonth = String(row[4] || '').trim();
      if (month && !rowMonth.startsWith(month)) continue;

      // Format created date
      let createdDate = '';
      if (row[1] instanceof Date) {
        createdDate = Utilities.formatDate(row[1], Session.getScriptTimeZone(), 'yyyy-MM-dd HH:mm');
      } else {
        createdDate = String(row[1] || '').substring(0, 16);
      }

      payouts.push({
        payoutId:      row[0],
        createdDate:   createdDate,
        invoiceAmount: row[2] !== '' && row[2] !== null && row[2] !== undefined ? row[2] : '',
        payoutAmount:  row[3] !== '' && row[3] !== null && row[3] !== undefined ? row[3] : '',
        payoutMonth:   row[4] || '',
        notes:         row[5] || '',
        enteredBy:     row[6] || '',
        hpCompany:     row[7] || ''
      });
    }

    // Latest first
    payouts.reverse();
    return { success: true, payouts };
  } catch(e) { return { success: false, message: e.message }; }
}

      if (!row[1]) continue; // skip rows without invoice no
      const hp = String(row[12] || '').trim();
      if (!hp) continue;
      if (!stats[hp]) stats[hp] = { count: 0, totalDisbursed: 0, totalPayout: 0, refBreakdown: {} };
      stats[hp].count++;
      const disb = parseFloat(row[13]);
      if (!isNaN(disb)) stats[hp].totalDisbursed += disb;
      const refCust = String(row[8] || '').trim() || 'Unknown'; // col I: Ref Customer
      stats[hp].refBreakdown[refCust] = (stats[hp].refBreakdown[refCust] || 0) + 1;
    }

    // --- Aggregate from financierpayout sheet (col H = HPCompany, col D = PayoutAmount) ---
    const paySheet = getSheet('financierpayout');
    const payData  = paySheet.getDataRange().getValues();

    for (let i = 0; i < payData.length; i++) {
      const row = payData[i];
      if (!row[0]) continue;
      if (String(row[0]).toLowerCase() === 'payoutid') continue;
      const hp  = String(row[7] || '').trim(); // col H: HPCompany
      if (!hp) continue;
      if (!stats[hp]) stats[hp] = { count: 0, totalDisbursed: 0, totalPayout: 0 };
      const payout = parseFloat(row[3]); // col D: PayoutAmount
      if (!isNaN(payout)) stats[hp].totalPayout += payout;
    }

    // Build result array sorted by count desc
    const result = Object.keys(stats).sort().map(function(hp) {
      const s = stats[hp];
      const pct = s.totalDisbursed > 0 ? ((s.totalPayout / s.totalDisbursed) * 100).toFixed(1) : '0.0';
      return {
        hpCompany:      hp,
        count:          s.count,
        totalDisbursed: s.totalDisbursed,
        totalPayout:    s.totalPayout,
        percentage:     parseFloat(pct),
        refBreakdown:   s.refBreakdown
      };
    });

    return { success: true, report: result };
  } catch(e) { return { success: false, message: e.message }; }
}

/**
 * Disbursement Comparison Report — admin only, password protected
 * Joins Data sheet (col BI = frame, col AG = disbursed) with
 * HSRP sheet (col G = frame, col N = disbursal) by Frame Number.
 * Returns: customerName, variant, frameNo, dataDisbursed, hsrpDisbursed, difference
 */
function getDisbursementComparison(sessionId, password) {
  try {
    const validation = validateSession(sessionId);
    if (!validation.success) return { success: false, message: validation.message || 'Session expired' };
    const user = validation.user;
    if (user.role !== 'admin') return { success: false, message: 'Access denied — admin only' };
    if (password !== CONFIG.DISBURSEMENT_REPORT_PASSWORD) return { success: false, message: 'Incorrect password' };

    // --- Read HSRP: build map of frameNo → hsrpDisbursed ---
    const hsrpSheet = getSheet('HSRP');
    const hsrpData  = hsrpSheet.getDataRange().getValues();
    const hsrpMap   = {}; // { frameNo: disbursalAmount }
    for (let i = 1; i < hsrpData.length; i++) {
      const row    = hsrpData[i];
      const frame  = String(row[6] || '').trim();   // col G: Frame No
      const disb   = parseFloat(row[13]);            // col N: Disbursal Amount
      if (frame) hsrpMap[frame] = isNaN(disb) ? '' : disb;
    }

    // --- Read Data sheet ---
    const dataSheet = getSheet(CONFIG.SHEETS.DATA);
    const data      = dataSheet.getDataRange().getValues();
    const rows      = [];

    for (let i = 1; i < data.length; i++) {
      const row      = data[i];
      if (!row[0]) continue;
      if (String(row[64] || '') === 'CANCELLED') continue;

      const frame        = String(row[60] || '').trim();  // col BI: Frame Number
      if (!frame) continue;

      const customerName  = String(row[3]  || '').trim(); // col D
      const variant       = String(row[6]  || '').trim(); // col G
      const dataDisbursed = parseFloat(row[32]);           // col AG: Disbursed Amount
      const dataDisb      = isNaN(dataDisbursed) ? '' : dataDisbursed;
      if (dataDisb === '') continue; // skip rows with no disbursement amount in Data sheet
      const hsrpDisb      = hsrpMap.hasOwnProperty(frame) ? hsrpMap[frame] : null; // null = not found in HSRP
      const diff          = (dataDisb !== '' && hsrpDisb !== null && hsrpDisb !== '')
                            ? (parseFloat(hsrpDisb) - parseFloat(dataDisb)) : null;

      rows.push({
        customerName,
        variant,
        frameNo:       frame,
        dataDisbursed: dataDisb,
        hsrpDisbursed: hsrpDisb,
        difference:    diff,
        matched:       diff !== null && diff === 0
      });
    }

    return { success: true, rows };
  } catch(e) { return { success: false, message: e.message }; }
}

/**
 * Helper: Format date only (no time)
 * Keep this if not already present
 */
function formatDateOnly(dateValue) {
  if (!dateValue) return '';
  try {
    if (typeof dateValue === 'string') {
      return dateValue.split(' ')[0];
    }
    const d = new Date(dateValue);
    return Utilities.formatDate(d, 'Asia/Kolkata', 'dd-MMM-yyyy');
  } catch (e) {
    return dateValue.toString().split(' ')[0];
  }
}



/**
 * Get Vehicle Mileage Dashboard
 * Shows: Current wallet balance, last mileage for each vehicle, monthly stats
 */
function getVehicleMileageDashboard(sessionId) {
  try {
    const validation = validateSession(sessionId);
    if (!validation.success) {
      return validation;
    }
    
    const user = validation.user;
    
    // Only operator, accounts, and admin can access
    if (user.role !== 'operator' && user.role !== 'accounts' && user.role !== 'admin') {
      return {
        success: false,
        message: 'Access denied. Only Operator, Accounts, and Admin can access this feature.'
      };
    }
    
    // Get HPCL Wallet balance
    const walletSheet = getSheet('HPCLWallet');
    const walletData = walletSheet.getDataRange().getValues();
    
    let currentBalance = 0;
    if (walletData.length > 1) {
      // Get last balance (Column E, index 4)
      currentBalance = parseFloat(walletData[walletData.length - 1][4]) || 0;
    }
    
    // Get Vehicle Mileage data
    const mileageSheet = getSheet('VehicleMileage');
    const mileageData = mileageSheet.getDataRange().getValues();
    
    // Calculate stats for each vehicle
    const vehicles = ['BMW X1', 'Kia Sonet', 'TATA 407'];
    const vehicleStats = {};
    
    const now = new Date();
    const thisMonthStart = new Date(now.getFullYear(), now.getMonth(), 1);
    
    for (let v = 0; v < vehicles.length; v++) {
      const vehicleName = vehicles[v];
      
      let lastKm = 0;
      let lastMileage = 0;
      let lastDate = '';
      let monthlyFuel = 0;
      let monthlyDistance = 0;
      let totalEntries = 0;
      
      // Find last entry and monthly stats
      for (let i = mileageData.length - 1; i >= 1; i--) {
        const row = mileageData[i];
        const vehicle = row[2]; // Column C
        const date = new Date(row[1]); // Column B
        
        if (vehicle === vehicleName) {
          totalEntries++;
          
          // Get last entry
          if (lastKm === 0) {
            lastKm = row[4]; // Column E: New KM
            lastMileage = row[7]; // Column H: Mileage
            lastDate = formatDate(row[1]);
          }
          
          // Calculate monthly stats
          if (date >= thisMonthStart) {
            monthlyFuel += parseFloat(row[6]) || 0; // Column G: Liters
            monthlyDistance += parseFloat(row[5]) || 0; // Column F: Distance
          }
        }
      }
      
      const monthlyMileage = monthlyFuel > 0 ? (monthlyDistance / monthlyFuel).toFixed(2) : 0;
      
      vehicleStats[vehicleName] = {
        lastKm: lastKm,
        lastMileage: parseFloat(lastMileage).toFixed(2),
        lastDate: lastDate,
        monthlyFuel: monthlyFuel.toFixed(2),
        monthlyDistance: monthlyDistance.toFixed(0),
        monthlyMileage: monthlyMileage,
        totalEntries: totalEntries
      };
    }
    
    // Calculate monthly totals
    let totalMonthlyFuel = 0;
    let totalMonthlyCost = 0;
    
    for (let i = 1; i < mileageData.length; i++) {
      const row = mileageData[i];
      const date = new Date(row[1]);
      
      if (date >= thisMonthStart) {
        totalMonthlyFuel += parseFloat(row[6]) || 0; // Column G: Liters
        totalMonthlyCost += parseFloat(row[8]) || 0; // Column I: Cost
      }
    }
    
    return {
      success: true,
      dashboard: {
        walletBalance: currentBalance,
        lowBalanceAlert: currentBalance < 3000,
        vehicleStats: vehicleStats,
        monthlyTotals: {
          fuel: totalMonthlyFuel.toFixed(2),
          cost: totalMonthlyCost.toFixed(2)
        }
      }
    };
    
  } catch (error) {
    Logger.log('Error in getVehicleMileageDashboard: ' + error.toString());
    return {
      success: false,
      message: 'Error loading dashboard: ' + error.toString()
    };
  }
}

/**
 * Add fuel entry
 * Calculates mileage, updates HPCL wallet
 */
function addFuelEntry(sessionId, data) {
  try {
    const validation = validateSession(sessionId);
    if (!validation.success) {
      return validation;
    }
    
    const user = validation.user;
    
    if (user.role !== 'operator' && user.role !== 'accounts' && user.role !== 'admin') {
      return {
        success: false,
        message: 'Access denied'
      };
    }
    
    Logger.log('=== addFuelEntry ===');
    Logger.log('Vehicle: ' + data.vehicleName);
    Logger.log('New KM: ' + data.newKm);
    Logger.log('Liters: ' + data.liters);
    Logger.log('Cost: ' + data.cost);
    
    // Validate inputs
    if (!data.vehicleName || !data.newKm || !data.liters || !data.cost) {
      return {
        success: false,
        message: 'All fields are required'
      };
    }
    
    const newKm = parseFloat(data.newKm);
    const liters = parseFloat(data.liters);
    const cost = parseFloat(data.cost);
    
    // Get last KM for this vehicle
    const lastKmResult = getLastKmReading(sessionId, data.vehicleName);
    if (!lastKmResult.success) {
      return lastKmResult;
    }
    
    const previousKm = parseFloat(lastKmResult.lastKm) || 0;
    
    // Calculate distance
    const distance = newKm - previousKm;
    
    if (distance <= 0 && previousKm > 0) {
      return {
        success: false,
        message: 'New KM (' + newKm + ') must be greater than Previous KM (' + previousKm + ')'
      };
    }
    
    // Calculate mileage
    const mileage = distance > 0 ? (distance / liters).toFixed(2) : 0;
    
    // Calculate rate per liter
    const ratePerLiter = (cost / liters).toFixed(2);
    
    // Get current HPCL wallet balance
    const walletSheet = getSheet('HPCLWallet');
    const walletData = walletSheet.getDataRange().getValues();
    
    let currentBalance = 0;
    if (walletData.length > 1) {
      currentBalance = parseFloat(walletData[walletData.length - 1][4]) || 0;
    }
    
    // Check if sufficient balance
    if (currentBalance < cost) {
      return {
        success: false,
        message: 'Insufficient HPCL wallet balance. Current: ₹' + currentBalance.toFixed(2) + ', Required: ₹' + cost.toFixed(2)
      };
    }
    
    // Calculate new balance
    const newBalance = currentBalance - cost;
    
    // Get VehicleMileage sheet
    const mileageSheet = getSheet('VehicleMileage');
    const mileageData = mileageSheet.getDataRange().getValues();
    
    // Generate Entry ID
    const lastRow = mileageSheet.getLastRow();
    const entryId = 'VM' + String(lastRow).padStart(5, '0');
    
    const now = new Date();
    const entryDate = data.date || now;
    
    // Build mileage entry row
    const mileageRow = [
      entryId,                          // A: Entry ID
      entryDate,                        // B: Date
      data.vehicleName,                 // C: Vehicle Name
      previousKm,                       // D: Previous KM
      newKm,                            // E: New KM
      distance,                         // F: Distance
      liters,                           // G: Liters
      mileage,                          // H: Mileage
      cost,                             // I: Cost
      ratePerLiter,                     // J: Rate per Liter
      data.loadType || '',              // K: Load Type (TATA 407 only)
      data.loadOther || '',             // L: Load Other
      data.notes || '',                 // M: Notes
      user.name,                        // N: Operator Name
      newBalance,                       // O: HPCL Balance After
      now                               // P: Timestamp
    ];
    
    mileageSheet.appendRow(mileageRow);
    
    // Add wallet deduction entry
    const walletEntryId = 'TXN' + String(walletSheet.getLastRow()).padStart(5, '0');
    
    const walletRow = [
      walletEntryId,                    // A: Transaction ID
      entryDate,                        // B: Date
      'Deduction',                      // C: Type
      -cost,                            // D: Amount (negative for deduction)
      newBalance,                       // E: Balance
      data.vehicleName,                 // F: Vehicle
      entryId,                          // G: Entry ID (link)
      'Fuel: ' + liters + 'L @ ₹' + ratePerLiter + '/L', // H: Notes
      user.name                         // I: Updated By
    ];
    
    walletSheet.appendRow(walletRow);
    
    SpreadsheetApp.flush();
    
    Logger.log('✅ Fuel entry added successfully');
    Logger.log('Entry ID: ' + entryId);
    Logger.log('Mileage: ' + mileage + ' km/l');
    Logger.log('New wallet balance: ₹' + newBalance.toFixed(2));
    
    return {
      success: true,
      message: 'Fuel entry added successfully',
      data: {
        entryId: entryId,
        distance: distance,
        mileage: mileage,
        ratePerLiter: ratePerLiter,
        newBalance: newBalance,
        lowBalanceAlert: newBalance < 3000
      }
    };
    
  } catch (error) {
    Logger.log('Error in addFuelEntry: ' + error.toString());
    Logger.log('Stack: ' + error.stack);
    return {
      success: false,
      message: 'Error adding fuel entry: ' + error.toString()
    };
  }
}
 
/**
 * Get vehicle mileage history
 * Filter by vehicle and date range
 */
function getVehicleMileageHistory(sessionId, vehicleName, dateFrom, dateTo) {
  try {
    const validation = validateSession(sessionId);
    if (!validation.success) {
      return validation;
    }
    
    const user = validation.user;
    
    if (user.role !== 'operator' && user.role !== 'accounts' && user.role !== 'admin') {
      return {
        success: false,
        message: 'Access denied'
      };
    }
    
    const sheet = getSheet('VehicleMileage');
    const data = sheet.getDataRange().getValues();
    
    const results = [];
    
    // Parse date filters
    const fromDate = dateFrom ? new Date(dateFrom) : null;
    const toDate = dateTo ? new Date(dateTo) : null;
    
    if (fromDate) fromDate.setHours(0, 0, 0, 0);
    if (toDate) toDate.setHours(23, 59, 59, 999);
    
    // Filter records
    for (let i = 1; i < data.length; i++) {
      const row = data[i];
      const vehicle = row[2]; // Column C
      const date = new Date(row[1]); // Column B
      
      // Filter by vehicle
      if (vehicleName && vehicle !== vehicleName) {
        continue;
      }
      
      // Filter by date range
      if (fromDate && date < fromDate) continue;
      if (toDate && date > toDate) continue;
      
      results.push({
        entryId: row[0],
        date: formatDate(row[1]),
        vehicleName: row[2],
        previousKm: row[3],
        newKm: row[4],
        distance: row[5],
        liters: row[6],
        mileage: parseFloat(row[7]).toFixed(2),
        cost: parseFloat(row[8]).toFixed(2),
        ratePerLiter: parseFloat(row[9]).toFixed(2),
        loadType: row[10] || '',
        loadOther: row[11] || '',
        notes: row[12] || '',
        operatorName: row[13],
        balanceAfter: parseFloat(row[14]).toFixed(2)
      });
    }
    
    // Sort by date descending
    results.sort(function(a, b) {
      return new Date(b.date) - new Date(a.date);
    });
    
    // Calculate summary stats
    let totalDistance = 0;
    let totalLiters = 0;
    let totalCost = 0;
    
    for (let i = 0; i < results.length; i++) {
      totalDistance += parseFloat(results[i].distance) || 0;
      totalLiters += parseFloat(results[i].liters) || 0;
      totalCost += parseFloat(results[i].cost) || 0;
    }
    
    const averageMileage = totalLiters > 0 ? (totalDistance / totalLiters).toFixed(2) : 0;
    
    return {
      success: true,
      records: results,
      summary: {
        totalEntries: results.length,
        totalDistance: totalDistance.toFixed(0),
        totalLiters: totalLiters.toFixed(2),
        totalCost: totalCost.toFixed(2),
        averageMileage: averageMileage
      }
    };
    
  } catch (error) {
    Logger.log('Error in getVehicleMileageHistory: ' + error.toString());
    return {
      success: false,
      message: 'Error loading history: ' + error.toString()
    };
  }
}

    const rows = historyResult.records.map(function(record) {
      return [
        record.entryId,
        record.date,
        record.vehicleName,
        record.previousKm,
        record.newKm,
        record.distance,
        record.liters,
        record.mileage,
        record.cost,
        record.ratePerLiter,
        record.loadType,
        record.loadOther,
        record.notes,
        record.operatorName,
        record.balanceAfter
      ];
    });
    
    if (rows.length > 0) {
      newSheet.getRange(2, 1, rows.length, headers.length).setValues(rows);
    }
    
    // Add summary
    const summaryRow = newSheet.getLastRow() + 2;
    newSheet.getRange(summaryRow, 1).setValue('SUMMARY');
    newSheet.getRange(summaryRow, 1).setFontWeight('bold');
    
    newSheet.getRange(summaryRow + 1, 1).setValue('Total Entries:');
    newSheet.getRange(summaryRow + 1, 2).setValue(historyResult.summary.totalEntries);
    
    newSheet.getRange(summaryRow + 2, 1).setValue('Total Distance:');
    newSheet.getRange(summaryRow + 2, 2).setValue(historyResult.summary.totalDistance + ' km');
    
    newSheet.getRange(summaryRow + 3, 1).setValue('Total Fuel:');
    newSheet.getRange(summaryRow + 3, 2).setValue(historyResult.summary.totalLiters + ' L');
    
    newSheet.getRange(summaryRow + 4, 1).setValue('Total Cost:');
    newSheet.getRange(summaryRow + 4, 2).setValue('₹ ' + historyResult.summary.totalCost);
    
    newSheet.getRange(summaryRow + 5, 1).setValue('Average Mileage:');
    newSheet.getRange(summaryRow + 5, 2).setValue(historyResult.summary.averageMileage + ' km/l');
    
    // Auto-resize columns
    for (let i = 1; i <= headers.length; i++) {
      newSheet.autoResizeColumn(i);
    }
    
    // Freeze header row
    newSheet.setFrozenRows(1);
    
    // Share file
    const file = DriveApp.getFileById(newSS.getId());
    file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
    
    const fileUrl = newSS.getUrl();
    
    return {
      success: true,
      message: 'Excel file created with ' + rows.length + ' records',
      fileUrl: fileUrl
    };
    
  } catch (error) {
    Logger.log('Error in exportVehicleMileageToExcel: ' + error.toString());
    return {
      success: false,
      message: 'Error exporting: ' + error.toString()
    };
  }
}

// Update stock quantity (helper)
function updateInvStock(skuId, locationId, qtyDelta) {
  const sheet = getSheet('InvStock');
  const data = sheet.getDataRange().getValues();
  for (let i = 1; i < data.length; i++) {
    if (String(data[i][0]) === String(skuId) && String(data[i][1]) === String(locationId)) {
      const newQty = Math.max(0, (parseInt(data[i][2]) || 0) + qtyDelta);
      sheet.getRange(i + 1, 3).setValue(newQty);
      return;
    }
  }
  // New row
  sheet.appendRow([skuId, locationId, Math.max(0, qtyDelta)]);
}

function getBookingForIssue(params, sessionId) {
  const validation = validateSession(sessionId);
  if (!validation.success) return { success: false, message: validation.message || 'Session expired' };

  const searchType  = params.searchType  || 'receiptNo';
  const searchValue = (params.searchValue || '').toString().trim().toLowerCase();

  if (!searchValue) return { success: false, message: 'Search value is required' };

  // Data sheet column indices (0-based) — matches saveSales rowData order
  const COL_RECEIPT     = 0;   // A - Receipt No
  const COL_EXECUTIVE   = 1;   // B - Executive Name
  const COL_BOOKING_DT  = 2;   // C - Booking Date
  const COL_CUSTOMER    = 3;   // D - Customer Name
  const COL_MODEL       = 5;   // F - Model
  const COL_VARIANT     = 6;   // G - Variant
  const COL_DELIVERY    = 11;  // L - Delivery Date
  const COL_GUARD       = 12;  // M - Guard
  const COL_GRIPCOVER   = 13;  // N - Grip Cover
  const COL_SEATCOVER   = 14;  // O - Seat Cover
  const COL_MATIN       = 15;  // P - Matin
  const COL_TANKCOVER   = 16;  // Q - Tank Cover
  const COL_HANDLEHOOK  = 17;  // R - Handle Hook
  const COL_HELMET      = 18;  // S - Helmet
  const COL_ACCT_CHECK  = 21;  // V - Account Check (stored as 'Yes'/'No'/false)
  const COL_RAINCOVER   = 61;  // BJ - Rain Cover
  const COL_BUZZER      = 62;  // BK - Buzzer
  const COL_BACKREST    = 63;  // BL - Back Rest

  const sheet = getSheet(CONFIG.SHEETS.DATA);
  if (!sheet) return { success: false, message: 'Data sheet not found' };

  const data = sheet.getDataRange().getValues();

  function formatDate(val) {
    if (!val) return '';
    if (val instanceof Date) return Utilities.formatDate(val, Session.getScriptTimeZone(), 'yyyy-MM-dd');
    return val.toString();
  }

  function rowToBooking(row) {
    // accountCheck is initially boolean false, later set to 'Yes'/'No' string by accounts
    const rawAcct = row[COL_ACCT_CHECK];
    const accountCheck = (rawAcct === 'Yes' || rawAcct === true) ? 'Yes'
                       : (rawAcct === 'No') ? 'No' : '';
    return {
      receiptNo:     String(row[COL_RECEIPT]    || ''),
      customerName:  String(row[COL_CUSTOMER]   || ''),
      executiveName: String(row[COL_EXECUTIVE]  || ''),
      deliveryDate:  formatDate(row[COL_DELIVERY]),
      bookingDate:   formatDate(row[COL_BOOKING_DT]),
      model:         String(row[COL_MODEL]      || ''),
      variant:       String(row[COL_VARIANT]    || ''),
      accountCheck:  accountCheck,
      guard:      String(row[COL_GUARD]      || ''),
      gripcover:  String(row[COL_GRIPCOVER]  || ''),
      seatcover:  String(row[COL_SEATCOVER]  || ''),
      matin:      String(row[COL_MATIN]      || ''),
      tankcover:  String(row[COL_TANKCOVER]  || ''),
      handlehook: String(row[COL_HANDLEHOOK] || ''),
      helmet:     String(row[COL_HELMET]     || ''),
      raincover:  String(row[COL_RAINCOVER]  || ''),
      buzzer:     String(row[COL_BUZZER]     || ''),
      backrest:   String(row[COL_BACKREST]   || '')
    };
  }

  const matches = [];

  // Skip header row (i = 1)
  for (let i = 1; i < data.length; i++) {
    const row = data[i];
    if (!row[COL_RECEIPT]) continue; // skip empty rows

    let match = false;
    const deliveryVal = formatDate(row[COL_DELIVERY]);

    if (searchType === 'receiptNo') {
      match = row[COL_RECEIPT].toString().trim() === params.searchValue.toString().trim();
    } else if (searchType === 'customerName') {
      match = (row[COL_CUSTOMER] || '').toString().toLowerCase().includes(searchValue);
    } else if (searchType === 'executiveName') {
      match = (row[COL_EXECUTIVE] || '').toString().toLowerCase().includes(searchValue);
    } else if (searchType === 'deliveryDate') {
      match = deliveryVal === params.searchValue; // exact date yyyy-MM-dd
    }

    if (match) matches.push(rowToBooking(row));
  }

  if (matches.length === 0) {
    return { success: false, message: 'No booking found for the given search' };
  }

  if (matches.length === 1) {
    return { success: true, booking: matches[0] };
  }

  // Multiple results — return list for frontend picker
  return { success: true, bookings: matches };
}


function invIssueToBooking(params, sessionId) {
  const validation = validateSession(sessionId);
  if (!validation.success) return { success: false, message: validation.message || 'Session expired' };
  const user = validation.user;

  const receiptNo = params.receiptNo;
  const remarks   = params.remarks || '';
  let items;

  try {
    items = JSON.parse(params.items);
  } catch(e) {
    return { success: false, message: 'Invalid items data' };
  }
  if (!items || items.length === 0) return { success: false, message: 'No items provided' };

  // Look up variant and frame number from the booking (Data sheet)
  let variant = '';
  let frameNo  = '';
  try {
    const dataSheet = getSheet(CONFIG.SHEETS.DATA);
    const dataRows  = dataSheet.getDataRange().getValues();
    for (let i = 1; i < dataRows.length; i++) {
      if (String(dataRows[i][0]).trim() === String(receiptNo).trim()) {
        variant = String(dataRows[i][6]  || '');  // G  - Variant
        frameNo  = String(dataRows[i][60] || ''); // BI - Frame No
        break;
      }
    }
  } catch(e) {}

  // Process each item (each has its own skuId, qty, locationId)
  for (const item of items) {
    const q          = parseInt(item.qty) || 1;
    const skuId      = item.skuId;
    const locationId = item.locationId;
    if (!skuId || !locationId || !q) continue;

    // Check and deduct stock
    const stockSheet = getSheet('InvStock');
    const stockData  = stockSheet.getDataRange().getValues();
    let stockRowIndex = -1;
    let currentQty    = 0;
    for (let i = 1; i < stockData.length; i++) {
      if (String(stockData[i][0]) === String(skuId) && String(stockData[i][1]) === String(locationId)) {
        stockRowIndex = i + 1; // 1-based sheet row
        currentQty    = parseInt(stockData[i][2]) || 0;
        break;
      }
    }
    if (stockRowIndex === -1) {
      return { success: false, message: 'No stock record found for ' + getSkuName(skuId) + ' at selected location' };
    }
    if (currentQty < q) {
      return { success: false, message: 'Insufficient stock for ' + getSkuName(skuId) + '. Available: ' + currentQty };
    }
    stockSheet.getRange(stockRowIndex, 3).setValue(currentQty - q);

    // Log with correct column order: date first, then type
    logInvTransaction(
      'ISSUE', skuId, getSkuName(skuId),
      getLocationName(locationId), '',   // fromLocation, toLocation
      q, receiptNo,
      params.customerName || '', params.executiveName || '',
      '', '', '', remarks, user.name || user.email || '',
      variant, frameNo, ''
    );
  }

  return { success: true, message: 'Accessories issued successfully' };
}


function invOtcSale(params, sessionId) {

    }

}

function getInvTransactions(params, sessionId) {
  try {
const validation = validateSession(sessionId);
if (!validation.success) return { success: false, message: validation.message || 'Session expired' };
const user = validation.user;
    const { type, locationId, fromDate, toDate } = params;
    const sheet = getSheet('InvTransactions');
    const data = sheet.getDataRange().getValues();
    const transactions = [];
    
    for (let i = 0; i < data.length; i++) {
      const row = data[i];
      if (!row[0]) continue;
      // Skip header row (TxnId column header)
      if (String(row[0]).toString().toLowerCase() === 'txnid') continue;
      // Handle date: Google Sheets may return a Date object instead of string
      const rawDate = row[1];
      let rowDate = '';
      let displayDate = '';
      if (rawDate instanceof Date) {
        rowDate = Utilities.formatDate(rawDate, Session.getScriptTimeZone(), 'yyyy-MM-dd');
        displayDate = Utilities.formatDate(rawDate, Session.getScriptTimeZone(), 'yyyy-MM-dd HH:mm');
      } else {
        rowDate = String(rawDate).substring(0, 10);
        displayDate = String(rawDate).substring(0, 16);
      }
      if (type && row[2] !== type) continue;
      if (fromDate && rowDate < fromDate) continue;
      if (toDate && rowDate > toDate) continue;
      if (locationId) {
        // Get location name
        const locName = getLocationName(locationId);
        if (row[5] !== locName && row[6] !== locName) continue;
      }
      transactions.push({
        txnId: row[0], date: displayDate, type: row[2],
        skuId: row[3], skuName: row[4],
        fromLocation: row[5], toLocation: row[6], qty: row[7],
        receiptNo: row[8], customerName: row[9], executiveName: row[10],
        otcReceiptNo: row[11], saleAmount: row[12],
        remarks: row[14], enteredBy: row[15],
        variant: row[16], frameNo: row[17], invoiceNo: row[18]
      });
    }
    // Return latest first
    transactions.reverse();
    return { success: true, transactions };
  } catch(e) { return { success: false, message: e.message }; }
}

function cancelBooking(params, sessionId) {
  try {
    const validation = validateSession(sessionId);
    if (!validation.success) return { success: false, message: validation.message || 'Session expired' };
    const user = validation.user;
    if (user.role !== 'admin' && user.role !== 'sales') return { success: false, message: 'Access denied' };

    const receiptNo = (params.receiptNo || '').toString().trim();
    if (!receiptNo) return { success: false, message: 'Receipt number required' };

    const sheet = getSheet(CONFIG.SHEETS.DATA);
    const data  = sheet.getDataRange().getValues();

    for (let i = 1; i < data.length; i++) {
      if (String(data[i][0]).trim() === receiptNo) {
        const accountCheck = data[i][21];
        if (accountCheck === 'Yes') {
          return { success: false, message: 'Cannot cancel: Account Check is already "Yes"' };
        }
        // Write CANCELLED to column BM (index 64, 1-based col 65)
        sheet.getRange(i + 1, 65).setValue('CANCELLED');
        sheet.getRange(i + 1, 66).setValue(
          'Cancelled by ' + (user.name || user.username) + ' on ' +
          Utilities.formatDate(new Date(), Session.getScriptTimeZone(), 'yyyy-MM-dd HH:mm')
        );
        return { success: true, message: 'Booking ' + receiptNo + ' cancelled successfully' };
      }
    }
    return { success: false, message: 'Record not found: ' + receiptNo };
  } catch(e) { return { success: false, message: e.message }; }
}

/**
 * Return unique executive names from column B of Data sheet.
 * Sales role: returns only their own name.
 * Admin/accounts: returns all unique names.
 */
function getExecutiveNames(sessionId) {
  try {
    const validation = validateSession(sessionId);
    if (!validation.success) return { success: false, message: validation.message || 'Session expired' };
    const user = validation.user;

    const sheet = getSheet(CONFIG.SHEETS.DATA);
    const data = sheet.getDataRange().getValues();
    const nameSet = new Set();

    for (let i = 1; i < data.length; i++) {
      const row = data[i];
      if (!row[0]) continue;
      if (String(row[64] || '') === 'CANCELLED') continue;
      const name = String(row[1] || '').trim();
      if (!name) continue;

      if (user.role === 'sales') {
        const userName  = (user.name     || '').trim().toLowerCase();
        const userAlias = (user.username || '').trim().toLowerCase();
        if (name.toLowerCase() !== userName && name.toLowerCase() !== userAlias) continue;
      }

      nameSet.add(name);
    }

    const names = Array.from(nameSet).sort();
    return { success: true, names };
  } catch(e) { return { success: false, message: e.message }; }
}

function getPendingAccountCheck(params, sessionId) {
  try {
    const validation = validateSession(sessionId);
    if (!validation.success) return { success: false, message: validation.message || 'Session expired' };
    const user = validation.user;

    const month = (params.month || '').toString().trim(); // e.g. '2026-04'

    const sheet = getSheet(CONFIG.SHEETS.DATA);
    const data  = sheet.getDataRange().getValues();

    const stats = {}; // { execName: { count, records[] } }

    function fmtDate(val) {
      if (!val) return '';
      if (val instanceof Date) return Utilities.formatDate(val, Session.getScriptTimeZone(), 'yyyy-MM-dd');
      return String(val).substring(0, 10);
    }

    for (let i = 1; i < data.length; i++) {
      const row = data[i];
      if (!row[0]) continue;

      // Skip cancelled records
      if (String(row[64] || '') === 'CANCELLED') continue;

      const executive    = String(row[1] || '').trim();
      const accountCheck = row[21];
      const bookingDate  = fmtDate(row[2]);

      // Account check not done
      if (accountCheck === 'Yes') continue;

      // Month filter
      if (month && !bookingDate.startsWith(month)) continue;

      // Role: sales sees only own
      if (user.role === 'sales') {
        const userName = (user.name || user.username || '').trim();
        const userAlias = (user.username || '').trim();
        const execLower = executive.toLowerCase();
        if (execLower !== userName.toLowerCase() && execLower !== userAlias.toLowerCase()) continue;
      }

      if (!stats[executive]) stats[executive] = { count: 0, records: [] };
      stats[executive].count++;
      stats[executive].records.push({
        receiptNo:    String(row[0]),
        customerName: String(row[3] || ''),
        model:        (String(row[5] || '') + ' ' + String(row[6] || '')).trim(),
        bookingDate:  bookingDate,
        accountCheck: accountCheck === false || accountCheck === '' ? 'Blank' : String(accountCheck)
      });
    }

    return { success: true, stats };
  } catch(e) { return { success: false, message: e.message }; }
}

function getSalarySheet_(name) {
  return SpreadsheetApp.getActiveSpreadsheet().getSheetByName(name);
}

function genSalaryId_(prefix) {
  return prefix + Date.now() + Math.random().toString(36).substr(2,5).toUpperCase();
}

// Google Sheets auto-converts "2026-03" to a Date — convert back to YYYY-MM
function fmtMonth_(v) {
  if (!v && v !== 0) return '';
  if (v instanceof Date) return Utilities.formatDate(v, 'Asia/Kolkata', 'yyyy-MM');
  return String(v).trim();
}

function saveSalaryUpload(sessionId, month, records) {
  try {
    const session = validateSession(sessionId);
    if (!session.success) return { success: false, message: 'Not logged in' };
    const sheet = getSalarySheet_(CONFIG.SHEETS.SALARY_RECORDS);
    if (!sheet) return { success: false, message: 'SalaryRecords sheet not found' };
    const dateStr = Utilities.formatDate(new Date(), 'Asia/Kolkata', 'dd/MM/yyyy HH:mm');
    const enteredBy = session.user.name || session.user.username || '';
    let saved = 0;
    for (var i = 0; i < records.length; i++) {
      var rec = records[i];
      var recordId = genSalaryId_('SAL');
      sheet.appendRow([
        recordId, rec.type||'Salary-Bank', month, month,
        rec.empCode||'', rec.name||'', rec.uan||'', rec.paymentMode||'EPF',
        rec.gross||0, rec.presentDays||0, rec.basic||0, rec.hra||0,
        rec.conveyance||0, rec.medical||0, rec.education||0, rec.wash||0,
        rec.ot||0, rec.netSalary||0, rec.esic||0, rec.epf||0,
        rec.pt||0, rec.otherDeduction||0, rec.netPay||0, rec.netPay||rec.amount||0,
        rec.remarks||'', dateStr, enteredBy
      ]);
      // Force month columns (C=3, D=4) as plain text to prevent date auto-conversion
      var newRow = sheet.getLastRow();
      sheet.getRange(newRow, 3).setNumberFormat('@STRING@').setValue(month);
      sheet.getRange(newRow, 4).setNumberFormat('@STRING@').setValue(month);
      saved++;
    }
    return { success: true, message: saved + ' records saved', saved: saved };
  } catch(e) { return { success: false, message: e.toString() }; }
}
